565 
566         Partition text1Partition = result.getPartitionByName("text1").getFirst();

650     private boolean isEqualByteArray(byte[] left, byte[] right) {
651         if (left == null || right == null) {
652             logger.logDebug(() -> "at least one of the inputs was null: left: %s right: %s".formatted(Arrays.toString(left), Arrays.toString(right)));
653             return false;
654         }
655         if (left.length != right.length) {
656             logger.logDebug(() -> "Not equal! left length: %d right length: %d".formatted(left.length, right.length));
657             return false;
658         }
659         for (int i = 0; i < left.length; i++) {
660             if (left[i] != right[i]) {
661                 int finalI = i;
662                 int finalI1 = i;
663                 logger.logDebug(() -> "Not equal! at index %d left was: %d right was: %d".formatted(finalI, left[finalI1], right[finalI1]));
664                 return false;
665             }
666         }
667         return true;
668     }
