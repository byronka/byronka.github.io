559     /**
560      * There are two primary ways to send data in requests and responses.
561      * <ol>
562      * <li>
563      * Url encoding - this is what we have been doing so far, converting
564      *    some values to percent-encoded.
565      * </li>
566      * <li>
567      * Multipart - providing a boundary token of our choice, along with
568      *    a couple headers to indicate critical information, then a blank line,
569      *    then the raw content.
570      * </li>
571      *</ol>
572      * <p>
573      * This test will cause us to create a program that is able to read each
574      * section of the multipart format.  The interesting/difficult problem
575      * I'm anticipating is that we need to be able to read binary and ascii.
576      *</p>
577      * <p>
578      * See, we will only read until we encounter the boundary text, which might
579      * be "abc123".  So if we read byte-by-byte, as soon as we hit a byte that
580      * can be "a", we need to jump into the possibility that we may be looking
581      * at the beginning of "abc123".
582      * </p>
583      */
584     @Test
585     public void test_MultiPartForm_HappyPath() throws IOException {
586         byte[] multiPartData = makeTestMultiPartData();
587 
588         // This is the core of the test - here's where we'll process receiving a multipart data
589 
590         final Function<RequestLine, Function<Request, Response>> testHandler = (sl -> r -> {
591             if (r.body().asString("text1").equals("I am a value that is text") &&
592                     (r.body().asBytes("image_uploads"))[0] == 1 &&
593                     (r.body().asBytes("image_uploads"))[1] == 2 &&
594                     (r.body().asBytes("image_uploads"))[2] == 3
595             ) {
596                 return Response.htmlOk("<p>r was </p>");
597             } else {
598               return new Response(_404_NOT_FOUND);
599             }
600         });
601 
602         WebFramework wf = new WebFramework(context, default_zdt);
603         try (Server primaryServer = webEngine.startServer(es, wf.makePrimaryHttpHandler(testHandler))) {
604             try (Socket socket = new Socket(primaryServer.getHost(), primaryServer.getPort())) {
605                 try (var client = webEngine.startClient(socket)) {
606                     InputStream is = client.getInputStream();
607 
608                     // send a GET request
609                     client.sendHttpLine("POST /some_endpoint HTTP/1.1");
610                     client.sendHttpLine("Host: localhost:8080");
611                     client.sendHttpLine("Content-Type: multipart/form-data; boundary=i_am_a_boundary");
