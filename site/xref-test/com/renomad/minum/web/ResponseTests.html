12       * Two {@link Response} should be considered equal

14       */
15      @Test
16      public void testTwoResponsesEqual() {
17          Response response1 = Response.htmlOk("foo");
18          Response response2 = Response.htmlOk("foo");
19          assertEquals(response1, response2);
20      }
21  
22      @Test
23      public void testTwoResponsesNotEqual() {
24          Response response1 = Response.htmlOk("foo");
25          Response response2 = Response.htmlOk("bar");
26          assertFalse(response1.equals(response2));
27      }
28  
29      /**
30       * If we use two different {@link Response} as keys in a
31       * map, they will be treated as the same if they have
32       * the same content.  That is why here, I can use
33       * response1 as a key to get response2's value.
34       */
35      @Test
36      public void testUseResponseAsKey() {
37          var myMap = new HashMap<Response, Integer>();
38          Response response1 = Response.htmlOk("foo");
39          Response response2 = Response.htmlOk("foo");
40  
41          myMap.put(response1, 42);
42          myMap.put(response2, 88);
43  
44          assertEquals(myMap.get(response1), 88);
45      }
46  
47      @Test
48      public void testToString() {
49          Response response1 = Response.htmlOk("foo");
50          assertEquals(response1.toString(), "Response{statusCode=_200_OK, extraHeaders={Content-Type=text/html; charset=UTF-8}, body=[102, 111, 111]}");
51      }
