3   import java.io.*;
4   import java.net.InetSocketAddress;
5   import java.net.SocketAddress;
6   import java.nio.charset.StandardCharsets;
7   import java.util.function.Consumer;

9   
10  /**
11   * A fake of {@link ISocketWrapper} used in tests
12   */
13  public class FakeSocketWrapper implements ISocketWrapper {
14  
15      public Consumer<String> sendHttpLineAction;
16      public Supplier<String> getRemoteAddrAction;
17      public Supplier<SocketAddress> getRemoteAddrWithPortAction;
18      public OutputStream os;
19      public InputStream is;
20      public BufferedInputStream bufferedInputStream;
21  
22      public FakeSocketWrapper() {
23          is = new ByteArrayInputStream(new byte[0]);
24          os = new ByteArrayOutputStream();
25          this.getRemoteAddrAction = () -> "tester";
26          this.getRemoteAddrWithPortAction = () -> new InetSocketAddress("123.123.123.123", 1234);
27      }
28  
29      @Override
30      public void send(String msg) throws IOException {
31          os.write(msg.getBytes(StandardCharsets.UTF_8));
32      }
33  
34      @Override
35      public void send(byte[] bodyContents) throws IOException {
36          os.write(bodyContents);
37      }
38  
39      @Override
40      public void send(byte[] bodyContents, int off, int len) throws IOException {
41          os.write(bodyContents, off, len);
42      }
43  
44      @Override
45      public void send(int b) throws IOException {
46          os.write(b);
47      }
48  
49      @Override
50      public void sendHttpLine(String msg) {
51          sendHttpLineAction.accept(msg);
52      }
53  
54      @Override
55      public int getLocalPort() {
56          return 0;
57      }
58  
59      @Override
60      public SocketAddress getRemoteAddrWithPort() {
61          return getRemoteAddrWithPortAction.get();
62      }
63  
64      @Override
65      public String getRemoteAddr() {
66          return getRemoteAddrAction.get();
67      }
68  
69      @Override
70      public HttpServerType getServerType() {
71          return null;
72      }
73  
74      @Override
75      public void close() {}
76  
77      @Override
78      public InputStream getInputStream() {
79          if (bufferedInputStream == null) {
80              bufferedInputStream = new BufferedInputStream(this.is);
81          }
82          return bufferedInputStream;
83      }
84  
85      @Override
86      public String getHostName() {
87          return null;
88      }
89  
90      @Override
91      public String toString() {
92          return "fake socket wrapper";
93      }
94  
95      @Override
96      public void flush() {}
97  }
