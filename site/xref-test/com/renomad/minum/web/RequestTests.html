607      * zip along, changing our status based on encountering ampersands and equal
608      * signs.  we just need a way to somehow provide an inputstream or something like
609      * an input stream that, we can pull a byte at a time, and it in turn pulls
610      * a byte at a time from the socket.
611      * <br>
612      * UrlEncodedDataGetter.ready will return the next byte, or readAllBytes, predicated on whether the
613      * byte is *not* an ampersand.
614      */
615     @Test
616     public void testReadingAStreamingUrlEncoded() throws IOException {
617         byte[] dataBytes = "abc=123&foo=bar".getBytes(StandardCharsets.UTF_8);
618         var inputStream = new ByteArrayInputStream(dataBytes);
619         FakeSocketWrapper socketWrapper = new FakeSocketWrapper();
620         socketWrapper.is = inputStream;
621         byte result;
622         byte[] results;
623         List<String> keys = List.of("abc", "foo");
624         List<Character> firstByte = List.of('1', 'b');
625         List<byte[]> remainders = List.of(
626                 "23".getBytes(StandardCharsets.UTF_8),
627                 "ar".getBytes(StandardCharsets.UTF_8));
628         int index = 0;
629         IRequest request = makeRequest(List.of("content-length: " + dataBytes.length, "content-type: application/x-www-form-urlencoded"), socketWrapper);
630 
631         Iterable<UrlEncodedKeyValue> pairs = request.getUrlEncodedIterable();
632 
633         for (UrlEncodedKeyValue pair : pairs) {
634             String key = pair.getKey();
635             assertEquals(key, keys.get(index));
636             try (UrlEncodedDataGetter uedg = pair.getUedg()) {
637                 result = (byte) uedg.read();
638                 assertEquals((char) result, firstByte.get(index));
639                 results = uedg.readAllBytes();
640             }
641             assertEquals(new String(results, StandardCharsets.UTF_8), new String(remainders.get(index), StandardCharsets.UTF_8));
642             index++;
643         }

648     /**
649      * Similar to {@link #testReadingAStreamingUrlEncoded}, this will read the multipart form data
650      * as a stream.  The boundary is the value provided us in the header.  So
651      * all we need to do is to determine when we are at a boundary by the
652      * simplest way possible - create an array the size of the value we are
653      * seeking and moving it as a window along the data, halting when we
654      * have a match.
655      * <br>
656      * MultiPartFormDataGetter.read will return the next byte, or readAllBytes, predicated
657      * on whether we have not yet hit the window.  The window will scan ahead, so that
658      * the value returned to us is behind what we have read off the socket.
659      */
660     @Test
661     public void testReadingStreamingMultipart() throws IOException {
662         byte[] multipartDataBytes = makeTestMultiPartData();
663         var inputStream = new ByteArrayInputStream(multipartDataBytes);
664         FakeSocketWrapper socketWrapper = new FakeSocketWrapper();
665         socketWrapper.is = inputStream;
666         int index = 0;
667         var types = List.of("text/plain", "application/octet-stream");
668         var names = List.of("text1", "image_uploads");
669         var filenames = List.of("", "photo_preview.jpg");
670         var values = List.of("I am a value that is text".getBytes(StandardCharsets.UTF_8), new byte[]{1,2,3});
671         IRequest request = makeRequest(List.of("content-length: " + multipartDataBytes.length, "content-type: multipart/form-data; boundary=i_am_a_boundary"), socketWrapper);
672 
673         Iterable<StreamingMultipartPartition> partitions = request.getMultipartIterable();
674 
675         for (StreamingMultipartPartition partition : partitions) {
676             try (partition) {
677                 Headers headers1 = partition.getHeaders();
678                 String contentTypeHeader = headers1.valueByKey("content-type").getFirst();
679                 assertEquals(contentTypeHeader, types.get(index));
680                 String name = partition.getContentDisposition().getName();
681                 assertEquals(name, names.get(index));
682                 String filename = partition.getContentDisposition().getFilename();
683                 assertEquals(filename, filenames.get(index));
684                 byte[] bytes = partition.readAllBytes();
685                 assertEqualByteArray(bytes, values.get(index));
686                 index++;
687             }
688         }
689     }
