272      * Sending a photo of a cute kitty

274     @Test
275     public void test_ExtractBodyFromBytes_Image() throws IOException {
276         byte[] kittyImageBytes = Files.readAllBytes(Path.of("src/test/resources/kitty.jpg"));
277         final var baos = new ByteArrayOutputStream();
278         baos.write(
279                 """
280                  ------WebKitFormBoundaryGlzbZJMmR2xSuAaT\r
281                  Content-Disposition: form-data; name="person_id"\r
282                  \r
283                  1a3e2d86-639f-49f8-8278-4e769a4d7222\r
284                  ------WebKitFormBoundaryGlzbZJMmR2xSuAaT\r
285                  Content-Disposition: form-data; name="image_uploads"; filename="kitty.jpg"\r
286                  Content-Type: image/png\r
287                  \r
288                  """.getBytes(StandardCharsets.UTF_8));
289         baos.write(kittyImageBytes);
290         baos.write(
291                 """
292                 \r
293                 ------WebKitFormBoundaryGlzbZJMmR2xSuAaT--\r
294                 
295                 """.getBytes(StandardCharsets.UTF_8));
296         var bodyProcessor = new BodyProcessor(context);
297         byte[] byteArray = baos.toByteArray();
298         ByteArrayInputStream inputStream = new ByteArrayInputStream(byteArray);
299 
300         StopwatchUtils stopwatch = new StopwatchUtils().startTimer();
301         Body bodyResult = bodyProcessor.extractBodyFromInputStream(
302                 byteArray.length,
303                 "Content-Type: multipart/form-data; boundary=----WebKitFormBoundaryGlzbZJMmR2xSuAaT",
304                 inputStream);
305 
306         byte[] bytes = bodyResult.getPartitionByName("image_uploads").getFirst().getContent();
307         assertEquals(kittyImageBytes.length, bytes.length);
308         assertEquals(bodyResult.getBodyType(), BodyType.MULTIPART);
309         assertEqualByteArray(kittyImageBytes, bytes);
310         long timeTakenMillis = stopwatch.stopTimer();
311         logger.logDebug(() -> "Took " + timeTakenMillis + " milliseconds to process this multipart data");
312     }
