11  public class TemplateSectionTests {

13      @Test
14      public void test_MissingKeyAndSubstring() {
15          var templateSection = new TemplateSection(null, null, 0);
16          assertThrows(InvariantException.class,
17                  "Either the key or substring must exist",
18                  () -> templateSection.render(null));
19      }
20  
21      @Test
22      public void test_HavingKeyAndSubstring() {
23          var templateSection = new TemplateSection("key", "substring", 0);
24          assertThrows(InvariantException.class,
25                  "If this object has a substring, then it must not have a key",
26                  () -> templateSection.render(null));
27      }
28  
29      /**
30       * Since the incoming builder is empty, no processing takes place,
31       * it just returns the empty builder.
32       */
33      @Test
34      public void test_processSectionOutside_builderEmpty() {
35          var sb = new StringBuilder();
36          ArrayList<TemplateSection> templateSections = new ArrayList<>();
37  
38          StringBuilder result = TemplateProcessor.processSectionOutside(sb, templateSections, 0);
39  
40          assertTrue(sb == result);
41      }
42  
43      /**
44       * If there is content in the builder, then we'll add its content
45       * to the list of TemplateSections
46       */
47      @Test
48      public void test_processSectionOutside_builderNotEmpty() {
49          var sb = new StringBuilder().append("hello world");
50          ArrayList<TemplateSection> templateSections = new ArrayList<>();
51  
52          StringBuilder result = TemplateProcessor.processSectionOutside(sb, templateSections, 0);
53  
54          assertEquals(result.toString(), "");
55          assertEquals(templateSections.getFirst().key, "hello world");
56      }
57  
58      @Test
59      public void test_justArrivedInside() {
60          assertTrue(TemplateProcessor.justArrivedInside("hello {{ world }}", '{', 6));
61          assertFalse(TemplateProcessor.justArrivedInside("{", '{', 0));
62      }
63  
64      /**
65       * In this test, the indent does not get used, since
66       * it is only one line.
67       */
68      @Test
69      public void test_indenting_edgeCase_NoIndent() {
70          TemplateSection templateSection = new TemplateSection("abc", null, 5);
71          RenderingResult render = templateSection.render(Map.of("abc", "foo foo"));
72          assertEquals(render.renderedSection(), "foo foo");
73      }
74  
75      /**
76       * In this test, the indent is applied to the next line
77       */
78      @Test
79      public void test_indenting_edgeCase_WithIndent() {
80          TemplateSection templateSection = new TemplateSection("abc", null, 5);
81          RenderingResult render = templateSection.render(Map.of("abc", "foo foo\nbar bar"));
82          assertEquals(render.renderedSection(), "foo foo\n    bar bar");
83      }
