55      /**
56       * Demonstrates what is necessary to build out an inner template.  In this
57       * example, we have a "ul" element representing the totality of our
58       * outer template, and then we expect to build out a list of names which is
59       * our inner template.
60       * <br>
61       * After construction of the TemplateProcessor instances,
62       * we will first render out the internal template, then put that in the outer.
63       */
64      @Test
65      public void test_Template_SimpleInnerTemplate() {
66          // set up the templates
67          var innerTemplate = TemplateProcessor.buildProcessor("<li>{{ name }}</li>");
68          var outerTemplate = TemplateProcessor.buildProcessor("""
69                  <ul>
70                      {{ inner_template_goes_here }}
71                  </ul>
72                  """);
73  
74          // render the inner template
75          String renderedInnerTemplate = innerTemplate.renderTemplate(
76                  List.of(Map.of("name", "alice"), Map.of("name", "bob")));
77  
78          // merge that into the outer template, rendering the full final result
79          String finalResult = outerTemplate.renderTemplate(Map.of("inner_template_goes_here", renderedInnerTemplate));
80  
81          assertEquals(finalResult, """
82                  <ul>
83                      <li>alice</li>
84                      <li>bob</li>
85                  </ul>
86                  """);
87      }

198     /**
199      * A test for a large and complex template - has one template
200      * nested in the other, looped.
201      * <br>
202      * <pre>
203      * Current records, on a HP ProDesk 600 G1 using an Intel Core i5-4590 CPU at 3.30GHz running Windows 10
204      * ------------------------------------------------------------------------------------------------------
205      *
206      *   5000 milliseconds for 500,000 templates, about 100k per second
207      *
208      * </pre>
209      *
210      */
211     @Test
212     public void test_Templating_LargeComplex_Performance() {
213         int warmupIterations = 5;
214         int mainIterations = 5;
215 
216         // the expected result
217         String expectedFullOutput = fileUtils.readTextFile("src/test/webapp/templates/templatebenchmarks/expected_stock_output.html");
218 
219         // create all the processors we'll need
220         TemplateProcessor individualStockProcessor = TemplateProcessor.buildProcessor(fileUtils.readTextFile("src/test/webapp/templates/templatebenchmarks/individual_stock.html"));
221         TemplateProcessor stockPrices = TemplateProcessor.buildProcessor(fileUtils.readTextFile("src/test/webapp/templates/templatebenchmarks/stock_prices.html"));
222 
223         // define some stock data for testing
224         List<Stock> stocks = Stock.dummyItems();
225 
226         // create reusable maps
227         List<Map<String,String>> stockPricesList = new ArrayList<>();
228 
229         // fill the maps with data from stocks
230         for (int i = 0; i < stocks.size(); i++) {
231             Stock stock = stocks.get(i);
232             HashMap<String, String> stockPricesMap = new HashMap<>();
233             stockPricesMap.put("class", i % 2 == 1 ? "even" : "odd"); // the example I used start with odd, so ...
234             stockPricesMap.put("index", String.valueOf(i + 1));
235             stockPricesMap.put("symbol", stock.getSymbol());
236             stockPricesMap.put("url", stock.getUrl());
237             stockPricesMap.put("name", stock.getName());
238             stockPricesMap.put("price", String.valueOf(stock.getPrice()));
239             stockPricesMap.put("is_negative_change", stock.getChange() < 0 ? " class=\"minus\"" : "");
240             stockPricesMap.put("is_negative_ratio", stock.getRatio() < 0 ? " class=\"minus\"" : "");
241             stockPricesMap.put("change", String.valueOf(stock.getChange()));
242             stockPricesMap.put("ratio", String.valueOf(stock.getRatio()));
243             stockPricesList.add(stockPricesMap);
244         }
245 
246         logger.logDebug(() -> "STARTING WARMUP");
247         benchmark(warmupIterations, stockPrices, individualStockProcessor, expectedFullOutput, stockPricesList);
248 
249         logger.logDebug(() -> "STARTING MAIN");
250         benchmark(mainIterations, stockPrices, individualStockProcessor, expectedFullOutput, stockPricesList);
251         benchmark(mainIterations, stockPrices, individualStockProcessor, expectedFullOutput, stockPricesList);
252         benchmark(mainIterations, stockPrices, individualStockProcessor, expectedFullOutput, stockPricesList);
253     }
254 
255     private static void benchmark(int renderingCount, TemplateProcessor stockPrices, TemplateProcessor individualStockProcessor, String expectedFullOutput, List<Map<String, String>> stockPricesList) {
256         StopwatchUtils stopwatch = new StopwatchUtils().startTimer();
257         IntStream.range(0, renderingCount).boxed().parallel().forEach(x -> {
258             String individualStockValuesRendered = individualStockProcessor.renderTemplate(stockPricesList);
259             String resultantString = stockPrices.renderTemplate(Map.of("individual_stocks", individualStockValuesRendered));
260             assertEquals(expectedFullOutput, resultantString);
261         });
262         logger.logDebug(() -> String.format("processed %d templates in %d millis", renderingCount, stopwatch.stopTimer()));
263     }

283     /**
284      * A TDD-style test to ensure the processor being thread-safe.
285      * Tests concurrent rendering with multiple threads and different data
286      * to verify that the TemplateProcessor correctly handles concurrent access
287      * without data corruption or race conditions.
288      */
289     @Test
290     public void test_Template_Multi_Thread() {
291         TemplateProcessor templateProcessor = buildProcessor("Hello {{name}}");
292         int threadCount = 10;
293         var futures = new ArrayList<CompletableFuture<String>>();
294 
295         // Create multiple concurrent render tasks with different data
296         for (int i = 0; i < threadCount; i++) {
297             final int threadNum = i;
298             var future = CompletableFuture.supplyAsync(() ->
299                     templateProcessor.renderTemplate(Map.of("name", "thread_" + threadNum))
300             );
301             futures.add(future);
302         }
303 
304         // Verify all threads completed successfully with correct results
305         for (int i = 0; i < threadCount; i++) {
306             String result = futures.get(i).join();
307             assertEquals(result, "Hello thread_" + i);
308 
309         }
310     }
