22       One major concern is that actions that are handled within ActionQueue
23       need to have a way to make exceptions known to the outside world. It's
24       a problem with multi-threaded code - too easy to miss exceptions.
25       */
26      @Test
27      public void test_ActionQueue_ErrorHandling() {
28          String message = "This is a test of ActionQueue handling an error";
29          var aq = new ActionQueue("Test ActionQueue", context).initialize();
30  
31          aq.enqueue("This should immediately fail", () -> {
32              throw new RuntimeException(message);
33          });
34  
35          // unavoidable race condition - if I check logger's list of messages without
36          // waiting, I will definitely get there before actionqueue.
37          MyThread.sleep(50);
38          String loggedMessage = logger.findFirstMessageThatContains(message);
39          assertTrue(!loggedMessage.isBlank(),
40                  "logged message must include expected message.  What was logged: " + loggedMessage);
41      }
