28      @Test
29      public void test_Template_Basic() {
30          String template = "Hello {{name}}, I'm {{animal}}";
31          var myMap = Map.of("name", "byron", "animal", "cat");
32          TemplateProcessor tp = buildProcessor(template);
33  
34          String renderedTemplate = tp.renderTemplate(myMap);
35  
36          assertEquals(renderedTemplate, "Hello byron, I'm cat");
