15      /**
16       * Stepping through the basic behaviors of our {@link RingBuffer}
17       */
18      @Test
19      public void test_RingBuffer_HappyPath() {
20          var rb = new RingBuffer<>(4, Integer.class);
21  
22          // just one element in the buffer - the number 1.
23          rb.add(1);
24          compare(rb, setExpectedValues(null, null, null, 1));
25  
26          // The next item is the number 2.
27          rb.add(2);
28          compare(rb, setExpectedValues(null, null, 1, 2));
29  
30          // The next item is the number 3.
31          rb.add(3);
32          compare(rb, setExpectedValues(null, 1, 2, 3));
33  
34          // The next item is the number 4.
35          rb.add(4);
36          compare(rb, setExpectedValues(1, 2, 3, 4));
37  
38          // The next item is the number 5.  This one will overwrite the oldest element, 1.
39          rb.add(5);
40          compare(rb, setExpectedValues(2, 3, 4, 5));
41  
42          // The next item is the number 6.  This one will overwrite the second oldest element, 2.
43          rb.add(6);
44          compare(rb, setExpectedValues(3, 4, 5, 6));
45      }
46  
47      /**
48       * A helper method to set up expected results when the ringbuffer
49       * has four slots.
50       */
51      private static List<Integer> setExpectedValues(Integer a, Integer b, Integer c, Integer d) {
52          List<Integer> expected = new ArrayList<>();
53          expected.add(a);
54          expected.add(b);
55          expected.add(c);
56          expected.add(d);
57          return expected;
58      }
59  
60      /**
61       * Using the {@link RingBuffer}'s iterator, loop through, comparing each value
62       * to a list we provide of expected results.
63       */
64      private static void compare(RingBuffer<Integer> rb, List<Integer> expected) {
65          int index = 0;
66          for (Integer i : rb) {
67              Integer b = expected.get(index);
68              assertTrue(Objects.equals(i, b), "at index " + index + ", the value was " + b);
69              index += 1;
70          }
71      }

73      /**
74       * testing a utility to see whether we can find some data anywhere
75       * inside the RingBuffer
76       */
77      @Test
78      public void testContains() {
79          RingBuffer<Character> characters = new RingBuffer<>(4, Character.class);
80          characters.add('a');
81          characters.add('b');
82          characters.add('c');
83          characters.add('d');
84  
85          assertTrue(characters.contains(List.of('a')));
86          assertTrue(characters.contains(List.of('a', 'b')));
87          assertTrue(characters.contains(List.of('a', 'b', 'c', 'd')));
88          assertTrue(characters.contains(List.of('b', 'c', 'd')));
89          assertTrue(characters.contains(List.of('c', 'd')));
90          assertTrue(characters.contains(List.of('d')));
91          assertFalse(characters.contains(List.of('e')));
92          assertFalse(characters.contains(List.of('a', 'b', 'z')));
93  
94          var ex1 = assertThrows(UtilsException.class, () -> characters.contains(List.of()));
95          assertEquals(ex1.getMessage(), "expected a valid non-empty list to search for in the RingBuffer");
96          var ex2 = assertThrows(UtilsException.class, () -> characters.contains(null));
97          assertEquals(ex2.getMessage(), "expected a valid non-empty list to search for in the RingBuffer");
98      }

100     /**
101      * testing a utility to see whether we can find some data at a particular
102      * place inside the RingBuffer
103      */
104     @Test
105     public void testContainsAt() {
106         RingBuffer<Character> characters = new RingBuffer<>(4, Character.class);
107         characters.add('a');
108         characters.add('b');
109         characters.add('c');
110         characters.add('d');
111 
112         assertTrue(characters.containsAt(List.of('a'), 0));
113         assertTrue(characters.containsAt(List.of('b', 'c', 'd'), 1));
114         assertFalse(characters.containsAt(List.of('b', 'c', 'd'), 0));
115         assertTrue(characters.containsAt(List.of('c', 'd'), 2));
116         assertTrue(characters.containsAt(List.of('d'), 3));
117         assertFalse(characters.containsAt(List.of('e'), 3));
118         assertFalse(characters.containsAt(List.of('d'), 0));
119         var ex = assertThrows(UtilsException.class, () -> characters.containsAt(List.of(), 0));
120         assertEquals(ex.getMessage(), "expected a valid non-empty list to search for in the RingBuffer");
121         var ex2 = assertThrows(UtilsException.class, () -> characters.containsAt(null, 0));
122         assertEquals(ex2.getMessage(), "expected a valid non-empty list to search for in the RingBuffer");
123         var ex3 = assertThrows(UtilsException.class, () -> characters.containsAt(List.of('a'), -1));
124         assertEquals(ex3.getMessage(), "expected an index greater than zero and less-than-or-equal to the last index of the buffer (the limit minus one)");
125         var ex4 = assertThrows(UtilsException.class, () -> characters.containsAt(List.of('a'), 4));
126         assertEquals(ex4.getMessage(), "expected an index greater than zero and less-than-or-equal to the last index of the buffer (the limit minus one)");
127         var ex5 = assertThrows(UtilsException.class, () -> characters.containsAt(List.of('a'), 5));
128         assertEquals(ex5.getMessage(), "expected an index greater than zero and less-than-or-equal to the last index of the buffer (the limit minus one)");
129     }

131     /**
132      * If the iterator for the {@link RingBuffer} is asked to go past its
133      * limit, it should return a {@link java.util.NoSuchElementException}
134      */
135     @Test
136     public void testIteratingPastLimit() {
137         RingBuffer<Integer> integers = new RingBuffer<>(3, Integer.class);
138         integers.add(1);
139         integers.add(2);
140         integers.add(3);
141         Iterator<Integer> iterator = integers.iterator();
142         assertEquals(1, iterator.next());
143         assertEquals(2, iterator.next());
144         assertEquals(3, iterator.next());
145         assertThrows(NoSuchElementException.class, iterator::next);
146     }
