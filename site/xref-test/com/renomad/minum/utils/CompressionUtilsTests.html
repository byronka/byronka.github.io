30      /**
31       * The simplest-possible test to get a sense of how the GzipOutputStream works
32       */
33      @Test
34      public void gzipIntroductionTest() {
35          var bytes = new byte[]{1,2,3};
36          var expectedBytes = new byte[]{31, -117, 8, 0, 0, 0, 0, 0, 0, -1, 99, 100, 98, 6, 0, 29, -128, -68, 85, 3, 0, 0, 0};
37          var compressedBytes = gzipCompress(bytes);
38          assertEqualByteArray(compressedBytes, expectedBytes);
39      }

41      /**
42       * Here's a simple test to try compressing a text file
43       */
44      @Test
45      public void gzipTest_textFile() {
46          var bytes = gettysburgAddress.getBytes(StandardCharsets.UTF_8);
47          var compressedBytes = gzipCompress(bytes);
48          // if you want to write this to file, to check things, uncomment below.
49          //  try (var fos = new FileOutputStream("foo.zip")) {
50          //      fos.write(compressedBytes);
51          //  }
52          byte[] decompressedBytes = gzipDecompress(compressedBytes);
53          String decompressedString = new String(decompressedBytes, StandardCharsets.UTF_8);
54          assertEquals(decompressedString, gettysburgAddress);
55      }

57      /**
58       * Let's see what kind of timing and memory usage we get when running
59       * this in serial
60       */
61      @Test
62      public void gzipTest_textFile_Performance() {
63          StopwatchUtils stopwatchUtils = new StopwatchUtils();
64          StopwatchUtils stopwatchUtils1 = stopwatchUtils.startTimer();
65          for (int i = 0; i < 100; i++) {
66              var bytes = gettysburgAddress.getBytes(StandardCharsets.UTF_8);
67              var compressedBytes = gzipCompress(bytes);
68              byte[] decompressedBytes = gzipDecompress(compressedBytes);
69              String decompressedString = new String(decompressedBytes, StandardCharsets.UTF_8);
70              assertEquals(decompressedString, gettysburgAddress);
71          }
72          long millis = stopwatchUtils1.stopTimer();
73          logger.logDebug(() -> "Compression performance test took " + millis + " milliseconds");
74      }

76      /**
77       * How does the system handle if we ask it to compress many
78       * requests in parallel?
79       */
80      @Test
81      public void gzipTest_textFile_Performance_highly_parallel() {
82          StopwatchUtils stopwatchUtils = new StopwatchUtils();
83          StopwatchUtils stopwatchUtils1 = stopwatchUtils.startTimer();
84  
85          try (var executor = Executors.newVirtualThreadPerTaskExecutor()) {
86              IntStream.range(0, 100).forEach(i -> {
87                  executor.submit(() -> {
88                      var bytes = gettysburgAddress.getBytes(StandardCharsets.UTF_8);
89                      var compressedBytes = gzipCompress(bytes);
90                      byte[] decompressedBytes = gzipDecompress(compressedBytes);
91                      String decompressedString = new String(decompressedBytes, StandardCharsets.UTF_8);
92                      assertEquals(decompressedString, gettysburgAddress);
93                  });
94              });
95          }
96  
97          long millis = stopwatchUtils1.stopTimer();
98          logger.logDebug(() -> "Compression performance test took " + millis + " milliseconds");
99      }
