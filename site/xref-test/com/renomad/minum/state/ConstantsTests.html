18  /**
19   * The {@link Constants} class is where we store values that
20   * should not change during the course of the program. You
21   * might think that means it can be static, and you would
22   * be right.  However, by making it an object, it does
23   * give us some leeway in our testing - it frees us to
24   * have multiple instances of the whole program running
25   * with potentially varying constants.  That kind of
26   * flexibility is not possible without making nearly
27   * everything an instantiable class.
28   */
29  public class ConstantsTests {
30  
31  
32      static private Context context;
33      static private TestLogger logger;
34  
35      @BeforeClass
36      public static void init() {
37          context = TestFramework.buildTestingContext("ConstantsTests");
38          logger = (TestLogger)context.getLogger();
39      }
40  
41      @AfterClass
42      public static void cleanup() {
43          TestFramework.shutdownTestingContext(context);
44      }
45  
46      @Rule(order = Integer.MIN_VALUE)
47      public TestWatcher watchman = new TestWatcher() {
48          protected void starting(Description description) {
49              logger.test(description.toString());
50          }
51      };

53      /**
54       * There are a few "getProps" methods in the
55       * Constants class, which are there to enable
56       * converting strings to more sophisticated
57       * types.
58       * <br>
59       * One in particular is the overload that returns
60       * a list of strings.  Here, we'll test that it
61       * behaves as expected.
62       */
63      @Test
64      public void testGetProps_Array() {
65          List<String> extraMimeMappings = Constants.extractList(" a,b, c, d, foo bar   , biz", "");
66          assertEquals(extraMimeMappings, List.of("a","b","c","d", "foo bar","biz"));
67          assertEquals(Constants.extractList("", ""), List.of(""));
68      }
69  
70      /**
71       * In your testing, it may be necessary to inject your own choice
72       * of properties to the constants file.  In particular, it might be
73       * common to set the root of the database, or the port of the server.
74       */
75      @Test
76      public void testCustomProps() {
77          Properties properties = new Properties();
78          properties.setProperty("SERVER_PORT","1234");
79          properties.setProperty("DB_DIRECTORY","/some/directory/here");
80          var constants = new Constants(properties);
81          assertEquals(constants.serverPort, 1234);
82          assertEquals(constants.dbDirectory, "/some/directory/here");
83          // if a property does not exist, it will use defaults
84          assertEquals(constants.socketTimeoutMillis, 7000);
85      }
86  
87      @Test
88      public void testGettingConfiguredPropertiesFromFile_NothingFound() {
89          Properties props = Constants.getConfiguredProperties("foo");
90          assertEquals(props.size(), 0);
91      }
92  
93      @Test
94      public void test_convertLoggingStringsToEnums() {
95          List<String> logLevels = List.of("DEBUG");
96          List<LoggingLevel> result = Constants.convertLoggingStringsToEnums(logLevels);
97          assertEquals(result.toString(), List.of(LoggingLevel.DEBUG).toString());
98      }
99
