11  /**
12   * The {@link Constants} class is where we store values that
13   * should not change during the course of the program. You
14   * might think that means it can be static, and you would
15   * be right.  However, by making it an object, it does
16   * give us some leeway in our testing - it frees us to
17   * have multiple instances of the whole program running
18   * with potentially varying constants.  That kind of
19   * flexibility is not possible without making nearly
20   * everything an instantiable class.
21   */
22  public class ConstantsTests {
23  
24      /**
25       * There are a few "getProps" methods in the
26       * Constants class, which are there to enable
27       * converting strings to more sophisticated
28       * types.
29       * <br>
30       * One in particular is the overload that returns
31       * a list of strings.  Here, we'll test that it
32       * behaves as expected.
33       */
34      @Test
35      public void testGetProps_Array() {
36          List<String> extraMimeMappings = Constants.extractList(" a,b, c, d, foo bar   , biz", "");
37          assertEquals(extraMimeMappings, List.of("a","b","c","d", "foo bar","biz"));
38          assertEquals(Constants.extractList("", ""), List.of(""));
39      }
40  
41      /**
42       * In your testing, it may be necessary to inject your own choice
43       * of properties to the constants file.  In particular, it might be
44       * common to set the root of the database, or the port of the server.
45       */
46      @Test
47      public void testCustomProps() {
48          Properties properties = new Properties();
49          properties.setProperty("SERVER_PORT","1234");
50          properties.setProperty("DB_DIRECTORY","/some/directory/here");
51          var constants = new Constants(properties);
52          assertEquals(constants.serverPort, 1234);
53          assertEquals(constants.dbDirectory, "/some/directory/here");
54          // if a property does not exist, it will use defaults
55          assertEquals(constants.socketTimeoutMillis, 7000);
56      }
57  
58      @Test
59      public void testGettingConfiguredPropertiesFromFile_NothingFound() {
60          Properties props = Constants.getConfiguredProperties("foo");
61          assertEquals(props.size(), 0);
62      }
63  
64      @Test
65      public void test_convertLoggingStringsToEnums() {
66          List<String> logLevels = List.of("DEBUG");
67          List<LoggingLevel> result = Constants.convertLoggingStringsToEnums(logLevels);
68          assertEquals(result.toString(), List.of(LoggingLevel.DEBUG).toString());
69      }
70  
71      @Test
72      public void test_convertLoggingStringsToEnums_NegativeCase() {
73          List<String> logLevels = List.of("FUBAR");
74          List<LoggingLevel> result = Constants.convertLoggingStringsToEnums(logLevels);
75          assertEquals(result.toString(), List.of().toString());
76      }
77  }
