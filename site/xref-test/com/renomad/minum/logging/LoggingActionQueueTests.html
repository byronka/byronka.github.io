35      @Test
36      public void testGetQueueThread() {
37          var testQueue = new LoggingActionQueue("my test queue", context.getExecutorService(), context.getConstants());
38          testQueue.initialize();
39          MyThread.sleep(10);
40          assertEquals(testQueue.getQueueThread().getName(), "my test queue");
41      }
42  
43      @Test
44      public void testGetQueue() {
45          var testQueue = new LoggingActionQueue("my test queue", context.getExecutorService(), context.getConstants());
46          testQueue.initialize();
47          testQueue.enqueue("Printing a test comment", () -> {MyThread.sleep(20);System.out.println("This is a test");});
48          testQueue.enqueue("Printing a test comment", () -> {MyThread.sleep(20);System.out.println("This is a test");});
49  
50          assertEquals(testQueue.getQueue().peek().toString(), "Printing a test comment");
51      }
52  
53      @Test
54      public void testErrorWhileRunningAction() throws InterruptedException {
55          LinkedBlockingQueue<RunnableWithDescription> foo = new LinkedBlockingQueue<>();
56          foo.add(new RunnableWithDescription(() -> {
57              throw new RuntimeException("This is a test exception");
58          }, "Testing runAction"));
59          assertFalse(LoggingActionQueue.runAction(foo));
60      }
61      
62      @Test
63      public void testHappyPathWhileRunningAction() throws InterruptedException {
64          LinkedBlockingQueue<RunnableWithDescription> foo = new LinkedBlockingQueue<>();
65          foo.add(new RunnableWithDescription(() -> System.out.println("hello world"), "Testing runAction"));
66          assertTrue(LoggingActionQueue.runAction(foo));
67      }
