10  /**
11   * The {@link Constants} class is where we store values that
12   * should not change during the course of the program. You
13   * might think that means it can be static, and you would
14   * be right.  However, by making it an object, it does
15   * give us some leeway in our testing - it frees us to
16   * have multiple instances of the whole program running
17   * with potentially varying constants.  That kind of
18   * flexibility is not possible without making nearly
19   * everything an instantiable class.
20   */
21  public class ConstantsTests {
22  
23      /**
24       * There are a few "getProps" methods in the
25       * Constants class, which are there to enable
26       * converting strings to more sophisticated
27       * types.
28       * <br>
29       * One in particular is the overload that returns
30       * a list of strings.  Here, we'll test that it
31       * behaves as expected.
32       */
33      @Test
34      public void testGetProps_Array() {
35          List<String> extraMimeMappings = Constants.extractList(" a,b, c, d, foo bar   , biz", "");
36          assertEquals(extraMimeMappings, List.of("a","b","c","d", "foo bar","biz"));
37      }
38  
39      /**
40       * In your testing, it may be necessary to inject your own choice
41       * of properties to the constants file.  In particular, it might be
42       * common to set the root of the database, or the port of the server.
43       */
44      @Test
45      public void testCustomProps() {
46          Properties properties = new Properties();
47          properties.setProperty("SERVER_PORT","1234");
48          properties.setProperty("DB_DIRECTORY","/some/directory/here");
49          var constants = new Constants(properties);
50          assertEquals(constants.SERVER_PORT, 1234);
51          assertEquals(constants.DB_DIRECTORY, "/some/directory/here");
52          // if a property does not exist, it will use defaults
53          assertEquals(constants.USE_VIRTUAL, false);
54  
55      }
56  }
