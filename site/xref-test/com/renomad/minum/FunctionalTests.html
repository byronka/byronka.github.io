32      @BeforeClass
33      public static void init() throws IOException {
34          context = buildTestingContext("_integration_test");
35          new FullSystem(context).start();
36          new TheRegister(context).registerDomains();
37          logger = (TestLogger) context.getLogger();
38          context = context;
39          ft = new FunctionalTesting(context);
40      }
41  
42      @AfterClass
43      public static void cleanup() throws IOException {
44          // delay a sec so our system has time to finish before we start deleting files
45          MyThread.sleep(500);
46          context.getFileUtils().deleteDirectoryRecursivelyIfExists(Path.of(context.getConstants().DB_DIRECTORY), context.getLogger());
47          var fs = context.getFullSystem();
48          fs.close();
49          context.getLogger().stop();
50          context.getExecutorService().shutdownNow();
51      }

53      @Test
54      public void test() throws Exception {
55          context.getFullSystem().getWebFramework().addMimeForSuffix("png", "image/png");
56  
57          logger.test("Request a static png image that needed a mime type we just provided");
58          assertEquals(ft.get("moon.png").statusLine().status(), _200_OK);
59          assertEquals(ft.get("moon.png").headers().valueByKey("content-type"), List.of("image/png"));
60  
61          logger.test("Request a static file.  First time it gets loaded from disk... ");
62          assertEquals(ft.get("index.html").statusLine().status(), _200_OK);
63  
64          logger.test("Second time, it gets loaded from cache");
65          assertEquals(ft.get("index.html").statusLine().status(), _200_OK);
66  
67          logger.test("what if we ask for a file that doesn't exist?");
68          assertEquals(ft.get("DOES_NOT_EXIST.html").statusLine().status(), _404_NOT_FOUND);
69  
70          logger.test("grab the photos page unauthenticated. We should be able to view the photos.");
71          TestResponse photos = ft.get("photos");
72          assertEquals(photos.statusLine().status(), _200_OK);
73          var pNode = photos.searchOne(TagName.A, Map.of("href", "index.html"));
74          assertEquals(pNode.innerText(), "Index");
75  
76          logger.test("go to the page for registering a user, while unauthenticated.");
77          assertEquals(ft.get("register").statusLine().status(), _200_OK);
78  
79          logger.test("register a user");
80          var registrationResponse = ft.post("registeruser", "username=foo&password=bar");
81          assertEquals(registrationResponse.statusLine().status(), _303_SEE_OTHER);
82          assertEquals(registrationResponse.headers().valueByKey("location"), List.of("login"));
83  
84          logger.test("Go to the login page, unauthenticated");
85          assertEquals(ft.get("login").statusLine().status(), _200_OK);
86  
87          logger.test("login as the user we registered");
88          var response = ft.post("loginuser", "username=foo&password=bar");
89          var cookieValue = String.join(";", response.headers().valueByKey("set-cookie"));
90  
91          logger.test("try visiting the registration page while authenticated (should get redirected)");
92          List<String> authHeader = List.of("Cookie: " + cookieValue);
93          var registrationResponseAuthd = ft.post("registeruser", "username=foo&password=bar", authHeader);
94          assertEquals(registrationResponseAuthd.statusLine().status(), _303_SEE_OTHER);
95          assertEquals(registrationResponseAuthd.headers().valueByKey("location"), List.of("index"));
96  
97          logger.test("try visiting the login page while authenticated (should get redirected)");
98          assertEquals(ft.get("login", authHeader).statusLine().status(), _303_SEE_OTHER);
99  
100         logger.test("visit the page for uploading photos, authenticated");
101         HtmlParseNode uploadNodeFound1 = ft.get("upload", authHeader).searchOne(TagName.LABEL, Map.of("for", "image_uploads"));
102         assertEquals(uploadNodeFound1.innerText(), "Choose images to upload (PNG, JPG)");
103 
104         logger.test("upload some content, authenticated");
105         ft.post("upload", "image_uploads=123&short_descriptionbar=&long_description=foofoo", authHeader);
106 
107         logger.test("check out what's on the photos page now, unauthenticated");
108         TestResponse response1 = ft.get("photos");
109         var response2 = response1.searchOne(TagName.IMG, Map.of("alt", "photo alt text"));
110         String photoUrl = response2.tagInfo().attributes().get("src");
111         assertTrue(photoUrl.contains("photo?name="));
112 
113         logger.test("look at the contents of a particular photo, unauthenticated");
114         var photoResponse = ft.get(photoUrl);
115         assertTrue(photoResponse.body().asBytes().length > 2);
116 
117         logger.test("check out what's on the sample domain page, authenticated");
118         assertTrue(ft.get("index", authHeader).body().asString().contains("Enter a name"));
119         assertTrue(ft.get("formEntry", authHeader).body().asString().contains("Name Entry"));
120         assertEquals(ft.post("testform", "name_entry=abc", authHeader).statusLine().status(), _303_SEE_OTHER);
121 
122         logger.test("logout");
123         assertEquals(ft.get("logout", authHeader).statusLine().status(), _303_SEE_OTHER);
124 
125         logger.test("if we try to upload a photo unauth, we're prevented");
126         assertEquals(ft.post("upload", "foo=bar").statusLine().status(), _401_UNAUTHORIZED);
127 
128         logger.test("if we try to upload a name on the sampledomain auth, we're prevented");
129         assertEquals(ft.post("testform", "foo=bar").statusLine().status(), _401_UNAUTHORIZED);

131         logger.test("request a static asset");
132         TestResponse staticResponse = ft.get("main.css");
133         assertEquals(staticResponse.headers().contentType(), "content-type: text/css");
134         assertTrue(staticResponse.body().asString().contains("margin-left: 0;"));

139         try {
140             ft.get("a".repeat(context.getConstants().MAX_READ_LINE_SIZE_BYTES + 1));

142             // this is kind of a weird case I'm looking into.

144             // an exception is thrown here because of a connection reset. A
145             // connection reset happens when a host sends packets to a closed
146             // socket connection.  I debugged through here on the Windows machine
147             // and it's like nothing is wrong.

149             // so in the meantime, since I just need to cause the log to be entered,
150             // this try-catch that eats the exception will have to do.

154         // we have to wait for it to get through some processing before we check.

157         assertEquals(failureMsg, "in readLine, client sent more bytes than allowed.  Current max: 500");
