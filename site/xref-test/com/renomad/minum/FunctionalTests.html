272     /**
273      * This method asserts we see "from SetOfSws" in the logs.  The full
274      * context is to see messages like "http server removed (SocketWrapper for remote address: /127.0.0.1:53533) from SetOfSws. size: 0",
275      * which indicates the server is closing sockets as soon as they're finished.
276      * <br>
277      * This test was needed to guarantee the system will clean up its sockets, that is,
278      * it will not have a leak.
279      */
280     private static void checkForClosedSocket() {
281         MyThread.sleep(15);
282         assertTrue(logger.doesMessageExist("from SetOfSws", 15));
283     }
