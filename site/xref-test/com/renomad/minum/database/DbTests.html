78      @Test
79      public void test_GeneralCapability() throws IOException {
80          fileUtils.deleteDirectoryRecursivelyIfExists(foosDirectory, logger);
81          final var foos = range(1,10).mapToObj(x -> new Foo(x, x, "abc"+x)).toList();
82          final var db = new Db<Foo>(foosDirectory, context, INSTANCE);
83  
84          // make some files on disk
85          for (var foo : foos) {
86              db.write(foo);
87          }
88  
89          // check that the files are now there.
90          // note that since our minum.database is *eventually* synced to disk, we need to wait a
91          // (milli)second or two here for them to get onto the disk before we check for them.
92          MyThread.sleep(FINISH_TIME);
93          for (var foo : foos) {
94              assertTrue(Files.exists(foosDirectory.resolve(foo.getIndex() + Db.DATABASE_FILE_SUFFIX)));
95          }
96  
97          // rebuild some objects from what was written to disk
98          // note that since our minum.database is *eventually* synced to disk, we need to wait a
99          // (milli)second or two here for them to get onto the disk before we check for them.
100         MyThread.sleep(FINISH_TIME);
101         assertEqualsDisregardOrder(
102                 db.values().stream().map(Foo::toString).toList(),
103                 foos.stream().map(Foo::toString).toList());
104 
105         // change those files
106         final var updatedFoos = new ArrayList<Foo>();
107         for (var foo : db.values().stream().toList()) {
108             final var newFoo = new Foo(foo.index, foo.a + 1, foo.b + "_updated");
109             updatedFoos.add(newFoo);
110             db.update(newFoo);
111         }
112 
113         // rebuild some objects from what was written to disk
114         // note that since our minum.database is *eventually* synced to disk, we need to wait a
115         // (milli)second or two here for them to get onto the disk before we check for them.
116         MyThread.sleep(FINISH_TIME);
117         assertEqualsDisregardOrder(
118                 db.values().stream().map(Foo::toString).toList(),
119                 updatedFoos.stream().map(Foo::toString).toList());
120 
121         // delete the files
122         for (var foo : foos) {
123             db.delete(foo);
124         }
125 
126         // check that all the files are now gone
127         // note that since our minum.database is *eventually* synced to disk, we need to wait a
128         // (milli)second or two here for them to get onto the disk before we check for them.
129         MyThread.sleep(FINISH_TIME);
130         for (var foo : foos) {
131             assertFalse(Files.exists(foosDirectory.resolve(foo.getIndex() + Db.DATABASE_FILE_SUFFIX)));
132         }
133 
134         // give the action queue time to save files to disk
135         // then shut down.
136         db.stop();
137         fileUtils.deleteDirectoryRecursivelyIfExists(Path.of("out/simple_db"), logger);
138         MyThread.sleep(FINISH_TIME);
139     }

157         // what if the directory is missing when try to deserialize?
158         // note: this would only happen if, after instantiating our db,
159         // the directory gets deleted/corrupted.

161         // clear out the directory to start
162         fileUtils.deleteDirectoryRecursivelyIfExists(foosDirectory, logger);
163         final var db = new Db<Foo>(foosDirectory, context, INSTANCE);
164         MyThread.sleep(10);
165 
166         // create an empty file, to create that edge condition
167         final var pathToSampleFile = foosDirectory.resolve("1.ddps");
168         Files.createFile(pathToSampleFile);
169         db.loadDataFromDisk();
170         MyThread.sleep(10);
171         String existsButEmptyMessage = logger.findFirstMessageThatContains("file exists but");
172         assertEquals(existsButEmptyMessage, "1.ddps file exists but empty, skipping");
173 
174         // create a corrupted file, to create that edge condition

176         final var ex = assertThrows(RuntimeException.class, () -> db.loadDataFromDisk());

178 
179         fileUtils.deleteDirectoryRecursivelyIfExists(foosDirectory, logger);
180         db.loadDataFromDisk();
181         MyThread.sleep(10);
182         String directoryMissingMessage = logger.findFirstMessageThatContains("directory missing, adding nothing").replace('\\', '/');
183         assertTrue(directoryMissingMessage.contains("foos directory missing, adding nothing to the data list"));
184         db.stop();
185         MyThread.sleep(FINISH_TIME);
186     }

188     /**
189      * When this is looped a hundred thousand times, it takes 500 milliseconds to finish
190      * making the updates in memory.  It takes several minutes later for it to
191      * finish getting those changes persisted to disk.
192      * a million writes in 500 milliseconds means 2 million writes in one sec.
193      */
194     @Test
195     public void test_Performance() throws IOException {
196         // clear out the directory to start
197         fileUtils.deleteDirectoryRecursivelyIfExists(foosDirectory, logger);
198         final var db = new Db<Foo>(foosDirectory, context, INSTANCE);
199         MyThread.sleep(10);
200 
201         final var foos = new ArrayList<Foo>();
202 
203         // write the foos
204         for (int i = 0; i < 10; i++) {
205             final var newFoo = new Foo(i, i + 1, "original");
206             foos.add(newFoo);
207             db.write(newFoo);
208         }
209 
210         // change the foos
211         final var outerTimer = new StopwatchUtils().startTimer();
212         final var innerTimer = new StopwatchUtils().startTimer();
213 
214         final var newFoos = new ArrayList<Foo>();
215         for (var i = 1; i < 10; i++) {
216             newFoos.clear();
217                 /*
218                 loop through the old foos and update them to new values,
219                 creating a new list in the process.  There should only
220                 ever be 10 foos.
221                  */
222             for (var foo : foos) {
223                 final var newFoo = new Foo(foo.getIndex(), foo.a + 1, foo.b + "_updated");
224                 newFoos.add(newFoo);
225                 db.update(newFoo);
226             }
227         }
228         logger.logDebug(() -> "It took " + innerTimer.stopTimer() + " milliseconds to make the updates in memory");
229         db.stop(10, 20);
230         logger.logDebug(() -> "It took " + outerTimer.stopTimer() + " milliseconds to finish writing everything to disk");
231 
232         final var db1 = new Db<Foo>(foosDirectory, context, INSTANCE);
233         Collection<Foo> values = db1.values();
234         assertTrue(newFoos.containsAll(values));
235         db1.stop(10, 20);
236 
237         final var pathToIndex = foosDirectory.resolve("index.ddps");
238         // this file should not be empty, but we are making it empty
239         Files.writeString(pathToIndex,"");
240         var ex = assertThrows(RuntimeException.class, () -> new Db<Foo>(foosDirectory, context, INSTANCE));
241         // because the error message includes a path that varies depending on which OS, using regex to search.
242         assertTrue(RegexUtils.isFound("Exception while reading out.simple_db.foos.index.ddps in Db constructor",ex.getMessage()));
243         MyThread.sleep(FINISH_TIME);
244     }

246     /**
247      * Test read and write.
248      * <p>
249      *      The database is mainly focused on in-memory, with eventual
250      *      disk persistence.  With that in mind, the files are only
251      *      read from disk once - the first time they are needed.  From
252      *      there, all disk operations are writes.
253      * </p>
254      * <p>
255      *      This test will try writing and reading in various ways to exercise
256      *      many of the ways the database can be used.
257      * </p>
258      * <p>
259      *     Namely, the first load of data for any particular {@link Db} class
260      *     can occur when the database is asked to analyze data, or to
261      *     write, update, or delete.
262      * </p>
263      */
264     @Test
265     public void test_Db_Write_and_Read() throws IOException {
266         // in this test, we're stopping and starting our database
267         // over and over - a very unnatural activity.  We need to
268         // provide sleep time for the actions to finish before
269         // moving to the next step.
270         int stepDelay = 30;
271 
272         fileUtils.deleteDirectoryRecursivelyIfExists(foosDirectory, logger);
273         var db = new Db<Foo>(foosDirectory, context, INSTANCE);
274         MyThread.sleep(stepDelay);
275         Foo foo1 = new Foo(1, 2, "a");
276         Foo foo2 = new Foo(2, 3, "b");
277 
278         // first round - adding to an empty database
279         db.write(foo1);
280         var foos = db.values().stream().toList();
281         assertEquals(foos.size(), 1);
282         assertEquals(foos.get(0), foo1);
283         db.stop();
284         MyThread.sleep(stepDelay);
285 
286         // second round - adding to a database that has stuff on disk
287         var db2 = new Db<Foo>(foosDirectory, context, INSTANCE);
288         db2.write(foo2);
289         MyThread.sleep(stepDelay);
290 
291         var foos2 = db2.values().stream().toList();
292         assertEquals(foos2.size(), 2);
293         assertEquals(foos2.get(0), foo1);
294         assertEquals(foos2.get(1), foo2);
295         db2.stop();
296         MyThread.sleep(stepDelay);
297 
298         // third round - reading from a database that has yet to read from disk
299         var db3 = new Db<Foo>(foosDirectory, context, INSTANCE);
300         var foos3 = db3.values().stream().toList();
301         assertEquals(foos3.size(), 2);
302         assertEquals(foos3.get(0), foo1);
303         assertEquals(foos3.get(1), foo2);
304         db3.stop();
305         MyThread.sleep(stepDelay);
306 
307         // fourth round - deleting from a database that has yet to read from disk
308         var db4 = new Db<Foo>(foosDirectory, context, INSTANCE);
309         db4.delete(foo2);
310         var foos4 = db4.values().stream().toList();
311         assertEquals(foos4.size(), 1);
312         assertEquals(foos4.get(0), foo1);
313         db4.stop();
314         MyThread.sleep(stepDelay);
315 
316         // fifth round - updating an item in a database that has not yet read from disk
317         var db5 = new Db<Foo>(foosDirectory, context, INSTANCE);
318         var updatedFoo1 = new Foo(1, 42, "update");
319         db5.update(updatedFoo1);
320         var foos5 = db5.values().stream().toList();
321         assertEquals(foos5.size(), 1);
322         assertEquals(foos5.get(0), updatedFoo1);
323         db5.stop();
324         MyThread.sleep(stepDelay);
325 
326         // sixth round - if we delete, it will reset the next index to 1
327         var db6 = new Db<Foo>(foosDirectory, context, INSTANCE);
328         db6.delete(updatedFoo1);
329         var foos6 = db6.values().stream().toList();
330         assertEquals(foos6.size(), 0);
331         db6.stop();
332         MyThread.sleep(stepDelay);
333 
334         // at this point, we should receive a new index of 1
335         var db7 = new Db<Foo>(foosDirectory, context, INSTANCE);
336         db7.write(foo1);
337         var foos7 = db7.values().stream().toList();
338         assertEquals(foos7.size(), 1);
339         assertEquals(foos7.get(0), foo1);
340         db7.stop();
341         MyThread.sleep(stepDelay);
342     }

371     static class Foo extends DbData<Foo> implements Comparable<Foo> {
372 
373         private long index;
374         private final int a;
375         private final String b;
376 
377         public Foo(long index, int a, String b) {
378             this.index = index;
379             this.a = a;
380             this.b = b;
381         }
382 
383         static final Foo INSTANCE = new Foo(0,0,"");
384 
385         @Override
386         public String serialize() {
387             return serializeHelper(index, a, b);
388         }
389 
390         @Override
391         public Foo deserialize(String serializedText) {
392             final var tokens =  deserializeHelper(serializedText);
393             return new Foo(
394                     Integer.parseInt(tokens.get(0)),
395                     Integer.parseInt(tokens.get(1)),
396                     tokens.get(2)
397                     );
398         }
399 
400         @Override
401         public boolean equals(Object o) {
402             if (this == o) return true;
403             if (o == null || getClass() != o.getClass()) return false;
404             Foo foo = (Foo) o;
405             return index == foo.index && a == foo.a && Objects.equals(b, foo.b);
406         }
407 
408         @Override
409         public int hashCode() {
410             return Objects.hash(index, a, b);
411         }
412 
413         @Override
414         public long getIndex() {
415             return index;
416         }
417 
418         @Override
419         public void setIndex(long index) {
420             this.index = index;
421         }
422 
423         // implementing comparator just so we can assertEqualsDisregardOrder on a collection of these
424         @Override
425         public int compareTo(Foo o) {
426             return Long.compare( o.getIndex() , this.getIndex() );
427         }
428 
429         @Override
430         public String toString() {
431             return "Foo{" +
432                     "index=" + index +
433                     ", a=" + a +
434                     ", b='" + b + '\'' +
435                     '}';
436         }
437     }
