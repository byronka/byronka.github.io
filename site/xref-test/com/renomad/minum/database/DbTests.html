235     /**
236      * When this is looped a hundred thousand times, it takes 500 milliseconds to finish
237      * making the updates in memory.  It takes several minutes later for it to
238      * finish getting those changes persisted to disk.
239      * a million writes in 500 milliseconds means 2 million writes in one sec.
240      */
241     @Test
242     public void test_Performance() throws IOException {
243         int originalFooCount = 50;
244         int loopCount = 50;
245 
246         // clear out the directory to start
247         fileUtils.deleteDirectoryRecursivelyIfExists(foosDirectory);
248 
249         // build a Context without testlogger - testlogger would impact perf here
250         Context contextWithRegularLogger = buildTestingContextWithRegularLogger("db_perf_testing");
251         final var db = new Db<>(foosDirectory, contextWithRegularLogger, INSTANCE);
252         MyThread.sleep(10);
253 
254         final var foos = new ArrayList<Foo>();
255 
256         // write the foos
257         for (int i = 0; i < originalFooCount; i++) {
258             final var newFoo = new Foo(0, i + 1, "original");
259             foos.add(newFoo);
260             db.write(newFoo);
261         }
262 
263         // change the foos
264         final var outerTimer = new StopwatchUtils().startTimer();
265         final var innerTimer = new StopwatchUtils().startTimer();
266 
267         final var newFoos = new ArrayList<Foo>();
268         for (var i = 1; i < loopCount; i++) {
269             newFoos.clear();
270                 /*
271                 loop through the old foos and update them to new values,
272                 creating a new list in the process.  There should only
273                 ever be 10 foos.
274                  */
275             for (var foo : foos) {
276                 final var newFoo = new Foo(foo.getIndex(), foo.a + 1, foo.b + "_updated");
277                 newFoos.add(newFoo);
278                 db.write(newFoo);
279             }
280         }
281         logger.logDebug(() -> "It took " + innerTimer.stopTimer() + " milliseconds to make the updates in memory");
282         db.stop(100, 50);
283         logger.logDebug(() -> "It took " + outerTimer.stopTimer() + " milliseconds to finish writing everything to disk");
284 
285         final var db1 = new Db<>(foosDirectory, context, INSTANCE);
286         Collection<Foo> values = db1.values();
287         assertTrue(newFoos.containsAll(values));
288         db1.stop(100, 50);
289 
290         final var pathToIndex = foosDirectory.resolve("index.ddps");
291         // this file should not be empty, but we are making it empty
292         Files.writeString(pathToIndex,"");
293         var ex = assertThrows(DbException.class, () -> new Db<>(foosDirectory, context, INSTANCE));
294         // because the error message includes a path that varies depending on which OS, using regex to search.
295         assertTrue(RegexUtils.isFound("Exception while reading out.simple_db.foos.index.ddps in Db constructor",ex.getMessage()));
296         MyThread.sleep(FINISH_TIME);
297     }

299     /**
300      * Test read and write.
301      * <p>
302      *      The database is mainly focused on in-memory, with eventual
303      *      disk persistence.  With that in mind, the files are only
304      *      read from disk once - the first time they are needed.  From
305      *      there, all disk operations are writes.
306      * </p>
307      * <p>
308      *      This test will try writing and reading in various ways to exercise
309      *      many of the ways the database can be used.
310      * </p>
311      * <p>
312      *     Namely, the first load of data for any particular {@link Db} class
313      *     can occur when the database is asked to analyze data, or to
314      *     write, update, or delete.
315      * </p>
316      */
317     @Test
318     public void test_Db_Write_and_Read() {
319         // in this test, we're stopping and starting our database
320         // over and over - a very unnatural activity.  We need to
321         // provide sleep time for the actions to finish before
322         // moving to the next step.
323         int stepDelay = 40;
324 
325         for (int i = 0; i < 3; i++) {
326             fileUtils.deleteDirectoryRecursivelyIfExists(foosDirectory);
327             var db = new Db<>(foosDirectory, context, INSTANCE);
328             MyThread.sleep(stepDelay);
329             Foo foo1 = new Foo(0, 2, "a");
330             Foo foo2 = new Foo(0, 3, "b");
331 
332             // first round - adding to an empty database
333             db.write(foo1);
334             var foos = db.values().stream().toList();
335             assertEquals(foos.size(), 1);
336             assertEquals(foos.getFirst(), foo1);
337             db.stop();
338             MyThread.sleep(stepDelay);
339 
340             // second round - adding to a database that has stuff on disk
341             var db2 = new Db<>(foosDirectory, context, INSTANCE);
342             db2.write(foo2);
343             MyThread.sleep(stepDelay);
344 
345             var foos2 = db2.values().stream().toList();
346             assertEquals(foos2.size(), 2);
347             assertEquals(foos2.get(0), foo1);
348             assertEquals(foos2.get(1), foo2);
349             db2.stop();
350             MyThread.sleep(stepDelay);
351 
352             // third round - reading from a database that has yet to read from disk
353             var db3 = new Db<>(foosDirectory, context, INSTANCE);
354             var foos3 = db3.values().stream().toList();
355             assertEquals(foos3.size(), 2);
356             assertEquals(foos3.get(0), foo1);
357             assertEquals(foos3.get(1), foo2);
358             db3.stop();
359             MyThread.sleep(stepDelay);
360 
361             // fourth round - deleting from a database that has yet to read from disk
362             var db4 = new Db<>(foosDirectory, context, INSTANCE);
363             db4.delete(foo2);
364             var foos4 = db4.values().stream().toList();
365             assertEquals(foos4.size(), 1);
366             assertEquals(foos4.getFirst(), foo1);
367             db4.stop();
368             MyThread.sleep(stepDelay);
369 
370             // fifth round - updating an item in a database that has not yet read from disk
371             var db5 = new Db<>(foosDirectory, context, INSTANCE);
372             var updatedFoo1 = new Foo(1, 42, "update");
373             db5.write(updatedFoo1);
374             var foos5 = db5.values().stream().toList();
375             assertEquals(foos5.size(), 1);
376             assertEquals(foos5.getFirst(), updatedFoo1);
377             db5.stop();
378             MyThread.sleep(stepDelay);
379 
380             // sixth round - if we delete, it will reset the next index to 1
381             var db6 = new Db<>(foosDirectory, context, INSTANCE);
382             db6.delete(updatedFoo1);
383             var foos6 = db6.values().stream().toList();
384             assertEquals(foos6.size(), 0);
385 
386             Foo newData = db6.write(new Foo(0, 1, "new data"));
387             assertEquals(newData.index, 1L);
388 
389             db6.delete(newData);
390 
391             db6.stop();
392             MyThread.sleep(stepDelay);
393         }
394     }

699     public static class Foo extends DbData<Foo> implements Comparable<Foo> {
700 
701         private long index;
702         private final int a;
703         private final String b;
704 
705         public Foo(long index, int a, String b) {
706             this.index = index;
707             this.a = a;
708             this.b = b;
709         }
710 
711         static final Foo INSTANCE = new Foo(0,0,"");
712 
713         @Override
714         public String serialize() {
715             return serializeHelper(index, a, b);
716         }
717 
718         @Override
719         public Foo deserialize(String serializedText) {
720             final var tokens =  deserializeHelper(serializedText);
721             return new Foo(
722                     Integer.parseInt(tokens.get(0)),
723                     Integer.parseInt(tokens.get(1)),
724                     tokens.get(2)
725                     );
726         }
727 
728         @Override
729         public boolean equals(Object o) {
730             if (this == o) return true;
731             if (o == null || getClass() != o.getClass()) return false;
732             Foo foo = (Foo) o;
733             return index == foo.index && a == foo.a && Objects.equals(b, foo.b);
734         }
735 
736         @Override
737         public int hashCode() {
738             return Objects.hash(index, a, b);
739         }
740 
741         @Override
742         public long getIndex() {
743             return index;
744         }
745 
746         @Override
747         public void setIndex(long index) {
748             this.index = index;
749         }
750 
751         // implementing comparator just so we can assertEqualsDisregardOrder on a collection of these
752         @Override
753         public int compareTo(Foo o) {
754             return Long.compare( o.getIndex() , this.getIndex() );
755         }
756 
757         @Override
758         public String toString() {
759             return "Foo{" +
760                     "index=" + index +
761                     ", a=" + a +
762                     ", b='" + b + '\'' +
763                     '}';
764         }
765     }
