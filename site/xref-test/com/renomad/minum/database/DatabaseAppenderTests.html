46      /**
47       * When the appender puts more lines into the currentAppendLog
48       * than the "maxAppendCount", it will save off the file to
49       * a new file in the appendLogDirectory with a name from the
50       * current date-time stamp.
51       */
52      @Test
53      public void test_SavingOffToNewFile() throws IOException {
54          fileUtils.deleteDirectoryRecursivelyIfExists(foosDirectory.resolve("test_SavingOffToNewFile"));
55          fileUtils.makeDirectory(foosDirectory.resolve("test_SavingOffToNewFile"));
56          var da = new DatabaseAppender(foosDirectory.resolve("test_SavingOffToNewFile"), context, fileUtils);
57          da.maxAppendCount = 5;
58  
59          da.appendCount = 4;
60  
61          String result1 = da.appendToDatabase(DatabaseChangeAction.UPDATE, "THIS IS A TEST");
62          assertTrue(result1.isEmpty(), "Result must be empty");
63  
64          da.appendCount = 5;
65  
66          String result2 = da.appendToDatabase(DatabaseChangeAction.UPDATE, "THIS IS A TEST");
67          assertFalse(result2.isEmpty(), "Result must not be empty");
68  
69          da.appendCount = 6;
70  
71          // need to have a little break here so we don't end up creating a new
72          // append log named with the same timestamp
73          MyThread.sleep(5);
74  
75          String result3 = da.appendToDatabase(DatabaseChangeAction.UPDATE, "THIS IS A TEST");
76          assertFalse(result3.isEmpty(), "Result must not be empty");
77      }
