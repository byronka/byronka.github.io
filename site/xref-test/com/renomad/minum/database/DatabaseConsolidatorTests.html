50      /**
51       * An exception should be thrown if the program cannot parse a line
52       * coming from the file to update (the second parameter)
53       */
54      @Test
55      public void testUpdatingData_EdgeCase_ParsingError() {
56          var ex = assertThrows(DbException.class, () -> DatabaseConsolidator.updateData("foo", List.of("bar"), List.of()));
57          assertEquals(ex.getMessage(), "Error parsing line in file.  File: foo line: bar");
58      }

60      /**
61       * An exception should be thrown if the program cannot parse the index
62       * value of the data
63       */
64      @Test
65      public void testUpdatingData_EdgeCase_ParsingErrorForIndex() {
66          var ex = assertThrows(DbException.class, () -> DatabaseConsolidator.updateData("foo", List.of("bar|biz|baz"), List.of()));
67          assertEquals(ex.getMessage(), "Failed to parse index from line in file. File: foo line: bar|biz|baz");
68      }

70      /**
71       * When we read a database change instruction string, the action must match an expected value, or
72       * else we will throw an exception
73       */
74      @Test
75      public void testParsingDatabaseChangeStrings_EdgeCase_InvalidAction() {
76          var ex = assertThrows(DbException.class, () -> DatabaseConsolidator.parseDatabaseChangeInstructionString("DONUT 1|biz|baz", "foo"));
77          assertEquals(ex.getMessage(), "Line in append-only log was missing an action (UPDATE or DELETE) in the first characters. Line was: DONUT 1|biz|baz");
78      }

80      /**
81       * The append-only logs in the appendlog directory must all be formatted as date-time
82       * strings, which we will parse and sort, so that we know which order to use when consolidating.
83       * Here, we demonstrate a situation where one of the files is not a valid filename
84       */
85      @Test
86      public void testParsingAppendLogs_EdgeCase_InvalidFilename() {
87          var ex = assertThrows(ParseException.class, () -> DatabaseConsolidator.convertFileListToDateList(new String[]{"NOT_A_DATE"}));
88          assertEquals(ex.getMessage(), "Unparseable date: \"NOT_A_DATE\"");
89      }
