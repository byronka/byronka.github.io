79      /**
80       * Test read and write.
81       * <p>
82       *      The database is mainly focused on in-memory, with eventual
83       *      disk persistence.  With that in mind, the files are only
84       *      read from disk once - the first time they are needed.  From
85       *      there, all disk operations are writes.
86       * </p>
87       * <p>
88       *      This test will try writing and reading in various ways to exercise
89       *      many of the ways the database can be used.
90       * </p>
91       * <p>
92       *     Namely, the first load of data for any particular {@link DbEngine2} class
93       *     can occur when the database is asked to analyze data, or to
94       *     write, update, or delete.
95       * </p>
96       */
97      @Test
98      public void test_Db_Write_and_Read() {
99          // in this test, we're stopping and starting our database
100         // over and over - a very unnatural activity.  We need to
101         // provide sleep time for the actions to finish before
102         // moving to the next step.
103         int stepDelay = 40;
104 
105         for (int i = 0; i < 3; i++) {
106             Path dbPathForTest = foosDirectory.resolve("test_Db_Write_and_Read");
107             fileUtils.deleteDirectoryRecursivelyIfExists(dbPathForTest);
108             var db = new DbEngine2<>(dbPathForTest, context, INSTANCE);
109             MyThread.sleep(stepDelay);
110             Foo foo1 = new Foo(0, 2, "a");
111             Foo foo2 = new Foo(0, 3, "b");
112 
113             // first round - adding to an empty database
114             db.write(foo1);
115             var foos = db.values().stream().toList();
116             assertEquals(foos.size(), 1);
117             assertEquals(foos.getFirst(), foo1);
118             db.stop();
119             MyThread.sleep(stepDelay);
120 
121             // second round - adding to a database that has stuff on disk
122             var db2 = new DbEngine2<>(dbPathForTest, context, INSTANCE);
123             db2.write(foo2);
124             MyThread.sleep(stepDelay);
125 
126             var foos2 = db2.values().stream().toList();
127             assertEquals(foos2.size(), 2);
128             assertEquals(foos2.get(0), foo1);
129             assertEquals(foos2.get(1), foo2);
130             db2.stop();
131             MyThread.sleep(stepDelay);
132 
133             // third round - reading from a database that has yet to read from disk
134             var db3 = new DbEngine2<>(dbPathForTest, context, INSTANCE);
135             var foos3 = db3.values().stream().toList();
136             assertEquals(foos3.size(), 2);
137             assertEquals(foos3.get(0), foo1);
138             assertEquals(foos3.get(1), foo2);
139             db3.stop();
140             MyThread.sleep(stepDelay);
141 
142             // fourth round - deleting from a database that has yet to read from disk
143             var db4 = new DbEngine2<>(dbPathForTest, context, INSTANCE);
144             db4.delete(foo2);
145             var foos4 = db4.values().stream().toList();
146             assertEquals(foos4.size(), 1);
147             assertEquals(foos4.getFirst(), foo1);
148             db4.stop();
149             MyThread.sleep(stepDelay);
150 
151             // fifth round - updating an item in a database that has not yet read from disk
152             var db5 = new DbEngine2<>(dbPathForTest, context, INSTANCE);
153             var updatedFoo1 = new Foo(1, 42, "update");
154             db5.write(updatedFoo1);
155             var foos5 = db5.values().stream().toList();
156             assertEquals(foos5.size(), 1);
157             assertEquals(foos5.getFirst(), updatedFoo1);
158             db5.stop();
159             MyThread.sleep(stepDelay);
160 
161             // sixth round - if we delete, it will reset the next index to 1
162             var db6 = new DbEngine2<>(dbPathForTest, context, INSTANCE);
163             db6.delete(updatedFoo1);
164             var foos6 = db6.values().stream().toList();
165             assertEquals(foos6.size(), 0);
166 
167             Foo newData = db6.write(new Foo(0, 1, "new data"));
168             assertEquals(newData.index, 1L);
169 
170             db6.delete(newData);
171 
172             db6.stop();
173             MyThread.sleep(stepDelay);
174         }
