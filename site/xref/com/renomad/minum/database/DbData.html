1   package com.renomad.minum.database;

3   /**
4    * An abstract data type meant to be used with {@link Db}
5    */
6   public abstract class DbData<T>{
7   
8       /**
9        * Serializes this object into a string representation.  It will be
10       * the values of this object as strings, encoded with URL encoding,
11       * separated by pipe symbols.
12       * @return this type serialized to a string - use {@link com.renomad.minum.utils.SerializationUtils#serializeHelper(Object[])}
13       */
14      protected abstract String serialize();
15  
16      /**
17       * deserializes the text back into an object.  See helper
18       * method {@link com.renomad.minum.utils.SerializationUtils#deserializeHelper(String)} to split a serialized
19       * string into tokens for rebuilding the object.  See
20       * also {@link #serialize()}
21       * @param serializedText the serialized string
22       * @return this type deserialized from a string
23       */
24      protected abstract T deserialize(String serializedText);
25  
26      /**
27       * We need an index so that each piece of data is distinct, even if it has the same data.
28       *
29       * @return an index for this data, used to name a file for this particular instance of data
30       */
31      protected abstract long getIndex();
32  
33      /**
34       * It is necessary for this method to exist because it is used
35       * by the {@link Db} code to add the new index into this data.
36       * <p>
37       *     Let me unpack that a bit.
38       * </p>
39       * <p>
40       *     The way our database works, it's expected that when you are creating
41       *     a new instance of data, you won't know its index yet, because that
42       *     is something the database manages for you.
43       * </p>
44       * <p>
45       *     The index is an {@link java.util.concurrent.atomic.AtomicLong} value
46       *     that allows us to create new data without worrying about race
47       *     conditions between threads (that is, we don't have to worry about
48       *     two threads accidentally adding the same index value to two different datas).
49       * </p>
50       * <p>
51       *     This value is also used as the name for the file of this data stored on disk.
52       * </p>
53       */
54      protected abstract void setIndex(long index);
55  
56      protected DbData() {}
57  }
