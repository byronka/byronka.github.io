1   package com.renomad.minum.database;

3   /**
4    * An abstract data type meant to be used with {@link Db}
5    */
6   public abstract class DbData<T>{
7   
8       /**
9        * Serializes this object into a string representation.  It will be
10       * the values of this object as strings, encoded with URL encoding,
11       * separated by pipe symbols.
12       * <p>
13       *     <em>An example:</em>
14       * </p>
15       * {@snippet :
16       *         import static com.renomad.minum.utils.SerializationUtils.serializeHelper;
17       *
18       *         public String serialize() {
19       *             return serializeHelper(index, a, b);
20       *         }
21       * }
22       * @return this type serialized to a string - use {@link com.renomad.minum.utils.SerializationUtils#serializeHelper(Object[])}
23       * @see #deserialize(String)
24       */
25      protected abstract String serialize();
26  
27      /**
28       * deserializes the text back into an object.  See helper
29       * method {@link com.renomad.minum.utils.SerializationUtils#deserializeHelper(String)} to split a serialized
30       * string into tokens for rebuilding the object.  See
31       * also {@link #serialize()}
32       *
33       * <p>
34       *     <em>An example: </em>
35       * </p>
36       * {@snippet :
37       *         import static com.renomad.minum.utils.SerializationUtils.deserializeHelper;
38       *
39       *         public Foo deserialize(String serializedText) {
40       *             final var tokens =  deserializeHelper(serializedText);
41       *             return new Foo(
42       *                     Integer.parseInt(tokens.get(0)),
43       *                     Integer.parseInt(tokens.get(1)),
44       *                     tokens.get(2)
45       *                     );
46       *         }
47       * }
48       * @param serializedText the serialized string
49       * @return this type deserialized from a string
50       * @see #serialize()
51       */
52      protected abstract T deserialize(String serializedText);
53  
54      /**
55       * We need an index so that each piece of data is distinct, even if it has the same data.
56       *
57       * @return an index for this data, used to name a file for this particular instance of data
58       */
59      protected abstract long getIndex();
60  
61      /**
62       * It is necessary for this method to exist because it is used
63       * by the {@link Db} code to add the new index into this data.
64       * <p>
65       *     Let me unpack that a bit.
66       * </p>
67       * <p>
68       *     The way our database works, it's expected that when you are creating
69       *     a new instance of data, you won't know its index yet, because that
70       *     is something the database manages for you.
71       * </p>
72       * <p>
73       *     The index is an {@link java.util.concurrent.atomic.AtomicLong} value
74       *     that allows us to create new data without worrying about race
75       *     conditions between threads (that is, we don't have to worry about
76       *     two threads accidentally adding the same index value to two different datas).
77       * </p>
78       * <p>
79       *     This value is also used as the name for the file of this data stored on disk.
80       * </p>
81       */
82      protected abstract void setIndex(long index);
83  
84      protected DbData() {}
85  }
