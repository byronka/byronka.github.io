1   package com.renomad.minum.security;

3   import java.io.IOException;
4   import java.util.Collection;
5   
6   /**
7    * Monitors the inmates who have misbehaved in our system.
8    * <p>
9    * a client who needs addressing will be stored in a map in this class for as
10   * long as required.  After they have served their time, they
11   * are released.
12   * </p>
13   * <p>
14   *     It is expected that users of this program will add ip addresses plus
15   *     a string for a complaint identifier, like "123.123.123.123_brute_forcing_login"
16   *     or "123.123.123.123_failed_login_too_many_times" or some such. In
17   *     particular, adding a suffix of "_vuln_seeking" will cause a client's
18   *     connection to be immediately dropped at the
19   *     onset - see code for {@link com.renomad.minum.web.WebFramework#dumpIfAttacker}
20   * </p>
21   * <p>
22   *     From a technical standpoint, this program stores strings paired
23   *     with timestamps, and removes those entries when hitting that time.
24   * </p>
25   */
26  public interface ITheBrig {
27      // Regarding the BusyWait - indeed, we expect that the while loop
28      // below is an infinite loop unless there's an exception thrown, that's what it is.
29      ITheBrig initialize();
30  
31      /**
32       * Kills the infinite loop running inside this class.
33       */
34      void stop() throws IOException;
35  
36      /**
37       * Put a client in jail for some infraction, for a specified time.
38       *
39       * @param clientIdentifier the client's ip address plus some complaint
40       *                         identifier, like 1.2.3.4_too_freq_downloads. One special
41       *                         string to use is "_vuln_seeking" - if this is appended
42       *                         to an ip address, then that client will have its
43       *                         connection immediately dropped upon connecting. For
44       *                         example, that would look like "123.123.123.123_vuln_seeking"
45       *
46       * @param sentenceDuration length of stay, in milliseconds
47       * @return whether we put this client in jail
48       */
49      boolean sendToJail(String clientIdentifier, long sentenceDuration);
50  
51      /**
52       * Return true if a particular client ip address is found
53       * in the list.
54       */
55      boolean isInJail(String clientIdentifier);
56  
57      /**
58       * Get the current list of ip addresses that have been
59       * judged as having carried out attacks on the system.
60       */
61      Collection<Inmate> getInmates();
62  }
