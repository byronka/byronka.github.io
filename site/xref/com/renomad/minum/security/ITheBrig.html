5   /**
6    * Monitors the inmates who have misbehaved in our system.
7    * <p>
8    * a client who needs addressing will be stored in a map in this class for as
9    * long as required.  After they have served their time, they
10   * are released.
11   * </p>
12   */
13  public interface ITheBrig {
14      // Regarding the BusyWait - indeed, we expect that the while loop
15      // below is an infinite loop unless there's an exception thrown, that's what it is.
16      ITheBrig initialize();
17  
18      /**
19       * Kills the infinite loop running inside this class.
20       */
21      void stop();
22  
23      /**
24       * Put a client in jail for some infraction, for a specified time.
25       *
26       * @param clientIdentifier the client's address plus some feature identifier, like 1.2.3.4_too_freq_downloads
27       * @param sentenceDuration length of stay, in milliseconds
28       * @return whether we put this client in jail
29       */
30      boolean sendToJail(String clientIdentifier, long sentenceDuration);
31  
32      /**
33       * Return true if a particular client ip address is found
34       * in the list.
35       */
36      boolean isInJail(String clientIdentifier);
37  
38      /**
39       * Get the current list of ip addresses that have been
40       * judged as having carried out attacks on the system.
41       */
42      List<Inmate> getInmates();
43  }
