109     public TheBrig(int sleepTime, Context context) {
110         this.es = context.getExecutorService();
111         this.constants = context.getConstants();
112         this.logger = context.getLogger();
113         Path dbDir = Path.of(constants.DB_DIRECTORY);
114         this.db = new Db<>(dbDir.resolve("the_brig"), context, Inmate.EMPTY);
115         this.clientKeys = this.db.values().stream().collect(Collectors.toMap(Inmate::getClientId, Inmate::getDuration));
116         this.sleepTime = sleepTime;
117     }

129     // Regarding the BusyWait - indeed, we expect that the while loop
130     // below is an infinite loop unless there's an exception thrown, that's what it is.
131     @Override
132     @SuppressWarnings({"BusyWait"})
133     public ITheBrig initialize() {
134         logger.logDebug(() -> "Initializing TheBrig main loop");
135         Callable<Object> innerLoopThread = () -> {
136             Thread.currentThread().setName("TheBrig");
137             myThread = Thread.currentThread();
138             while (true) {
139                 try {
140                     reviewCurrentInmates();
141                 } catch (InterruptedException ex) {
142 
143                     /*
144                     this is what we expect to happen.
145                     once this happens, we just continue on.
146                     this only gets called when we are trying to shut everything
147                     down cleanly
148                      */
149 
150                     if (constants.LOG_LEVELS.contains(LoggingLevel.DEBUG)) {
151                         System.out.printf("%s TheBrig is stopped.%n", TimeUtils.getTimestampIsoInstant());
152                     }
153                     Thread.currentThread().interrupt();
154                     return null;
155                 } catch (Exception ex) {
156                     System.out.printf("%s ERROR: TheBrig has stopped unexpectedly. error: %s%n",TimeUtils.getTimestampIsoInstant(), ex);
157                     throw ex;
158                 }
159             }
160         };
161         es.submit(innerLoopThread);
162         return this;
163     }
164 
165     private void reviewCurrentInmates() throws InterruptedException {
166         int size = clientKeys.size();
167         if (size > 0) {
168             logger.logTrace(() -> "TheBrig reviewing current inmates. Count: " + size);
169         }
170         var now = System.currentTimeMillis();
171         List<String> keysToRemove = new ArrayList<>();
172         // figure out which clients have paid their dues
173         for (var e : clientKeys.entrySet()) {
174             reviewForParole(now, keysToRemove, e);
175         }
176         for (var k : keysToRemove) {
177             logger.logTrace(() -> "TheBrig: removing " + k + " from jail");
178             List<Inmate> inmates1 = db.values().stream().filter(x -> x.clientId.equals(k)).toList();
179             mustBeTrue(inmates1.size() == 1, "There must be exactly one inmate found or there's a bug");
180             Inmate inmateToRemove = inmates1.get(0);
181             clientKeys.remove(k);
182             db.delete(inmateToRemove);
183         }
184         Thread.sleep(sleepTime);
185     }
186 
187     private void reviewForParole(long now, List<String> keysToRemove, Map.Entry<String, Long> e) {
188         if (e.getValue() < now) {
189             logger.logTrace(() -> "UnderInvestigation: " + e.getKey() + " has paid its dues and is getting released");
190             keysToRemove.add(e.getKey());
191         }
192     }

194     @Override
195     public void stop() {
196         logger.logDebug(() -> "TheBrig has been told to stop");
197         for (int i = 0; i < 10; i++) {
198             if (myThread != null) {
199                 logger.logDebug(() -> "TheBrig: Sending interrupt to thread");
200                 myThread.interrupt();
201                 return;
202             } else {
203                 MyThread.sleep(20);
204             }
205         }
206         throw new RuntimeException("TheBrig: Leaving without successfully stopping thread");
207     }

210     @Override
211     public void sendToJail(String clientIdentifier, long sentenceDuration) {
212         if (!constants.IS_THE_BRIG_ENABLED) {
213             return;
214         }
215         lock.lock(); // block threads here if multiple are trying to get in - only one gets in at a time
216         try {
217             logger.logDebug(() -> "TheBrig: Putting away " + clientIdentifier + " for " + sentenceDuration + " milliseconds");
218             clientKeys.put(clientIdentifier, System.currentTimeMillis() + sentenceDuration);
219             var existingInmates = db.values().stream().filter(x -> x.clientId.equals(clientIdentifier)).count();
220             mustBeTrue(existingInmates < 2, "count of inmates must be either 0 or 1, anything else is a bug");
221             if (existingInmates == 0) {
222                 Inmate newInmate = new Inmate(0L, clientIdentifier, System.currentTimeMillis() + sentenceDuration);
223                 db.write(newInmate);
224             }
225         } finally {
226             lock.unlock();
227         }
228 
229     }

231     @Override
232     public boolean isInJail(String clientIdentifier) {
233         if (!constants.IS_THE_BRIG_ENABLED) {
234             return false;
235         }
236         return clientKeys.containsKey(clientIdentifier);
