238     /**
239      * This overload allows you to specify that the contents of the
240      * properties file should be shown when it's read.
241      */
242     private Properties getConfiguredProperties() {
243         var properties = new Properties();
244         String fileName = "minum.config";
245         try (FileInputStream fis = new FileInputStream(fileName)) {
246             System.out.println(TimeUtils.getTimestampIsoInstant() +
247                     " found properties file at ./minum.config.  Loading properties");
248             properties.load(fis);
249         } catch (Exception ex) {
250             System.out.println(ConfigErrorMessage.getConfigErrorMessage());
251             System.exit(1);
252         }
253         return properties;
254     }

257     /**
258      * Given a list of strings representing logging levels,
259      * convert it to a list of enums.  Log levels are enumerated
260      * in {@link LoggingLevel}.
261      */
262     private List<LoggingLevel> convertLoggingStringsToEnums(List<String> logLevels) {
263         List<String> logLevelStrings = logLevels.stream().map(String::toUpperCase).toList();
264         List<LoggingLevel> enabledLoggingLevels = new ArrayList<>();
265         for (LoggingLevel t : LoggingLevel.values()) {
266             if (logLevelStrings.contains(t.name())) {
267                 enabledLoggingLevels.add(t);
268             }
269         }
270         return enabledLoggingLevels;
271     }
