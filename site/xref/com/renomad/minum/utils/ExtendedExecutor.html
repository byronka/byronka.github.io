8   /**
9    * This lets us capture the stack traces thrown in a thread,
10   * which the typical Executor does not.
11   */
12  public final class ExtendedExecutor extends ThreadPoolExecutor {
13  
14      public ExtendedExecutor(
15              int corePoolSize,
16              int maximumPoolSize,
17              long keepAliveTime,
18              TimeUnit unit,
19              BlockingQueue<Runnable> workQueue,
20              ThreadFactory threadFactory) {
21          super(corePoolSize, maximumPoolSize, keepAliveTime, unit, workQueue, threadFactory);
22      }
23  
24      @Override
25      protected void afterExecute(Runnable r, Throwable t) {
26          super.afterExecute(r, t);
27          if (t == null
28                  && r instanceof Future<?>
29                  && ((Future<?>)r).isDone()) {
30              try {
31                  ((Future<?>) r).get();
32              } catch (CancellationException ce) {
33                  t = ce;
34              } catch (ExecutionException ee) {
35                  t = ee.getCause();
36              } catch (InterruptedException ie) {
37                  // ignore/reset
38                  Thread.currentThread().interrupt();
39              }
40          }
41          if (t != null)
42              System.out.println(t);
43      }
44  
45      public static ExecutorService makeExecutorService(Constants constants) {
46          boolean useVirtualThreads = constants.USE_VIRTUAL;
47          if (constants.LOG_LEVELS.contains(LoggingLevel.DEBUG)) System.out.println(TimeUtils.getTimestampIsoInstant() + " use virtual threads? " + useVirtualThreads);
48          if (useVirtualThreads) {
49              // the following line is only usable with the virtual threads API, which
50              // is available on OpenJDK 19/20 in preview mode.
51              return Executors.newVirtualThreadPerTaskExecutor();
52          } else {
53              return new ExtendedExecutor(0, Integer.MAX_VALUE,
54                      60L, TimeUnit.SECONDS,
55                      new SynchronousQueue<>(),
56                      Executors.defaultThreadFactory());
57          }
58      }

60  }
