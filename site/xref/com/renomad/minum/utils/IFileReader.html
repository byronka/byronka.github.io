1   package com.renomad.minum.utils;
2   
3   import java.io.IOException;
4   import java.util.Map;
5   import java.util.concurrent.locks.ReentrantLock;
6   
7   /**
8    * An interface for the operations of reading a file.
9    * This primarily exists to make testing easier.
10   */
11  public interface IFileReader {
12  
13      /**
14       * Reads a file from disk.
15       * @throws com.renomad.minum.security.ForbiddenUseException if the requested path includes bad file patterns,
16       * mainly ones to escape from the intended directories (like ".." or "/", etc)
17       * @throws IllegalArgumentException if the path is blank
18       */
19      byte[] readFile(String path) throws IOException;
20  
21      /**
22       * This returns the {@link ReentrantLock} that is used around
23       * the {@link LRUCache} to allow access to the cache from
24       * other classes.  This is used in the {@link com.renomad.minum.web.WebFramework}
25       * class when accessing the cache for static files.
26       */
27      ReentrantLock getCacheLock();
28  
29      /**
30       * Returns the LRU (least-recently-used) cache that is
31       * filled when using {@link #readFile(String)}
32       */
33      Map<String, byte[]> getLruCache();
34  }
