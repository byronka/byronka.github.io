12      /**
13       * A helper method to reduce some of the boilerplate
14       * code when converting a list of bytes to an array.
15       * <p>
16       *     Often, we are gradually building up a list - the list takes
17       *     care of accommodating more elements as necessary. An
18       *     array, in contrast, is just a single size and doesn't
19       *     resize itself.  It's much less convenient to use, so we
20       *     more often use lists.
21       * </p>
22       */
23      public static byte[] byteListToArray(List<Byte> result) {
24          final var resultArray = new byte[result.size()];
25          for(int i = 0; i < result.size(); i++) {
26              resultArray[i] = result.get(i);
27          }
28          return resultArray;
29      }
