6   public final class SerializationUtils {
7   
8       private SerializationUtils() {
9           // not meant to be constructed.
10      }
11  
12      /**
13       * This is a helper that will encode the values you give it
14       * in preparation for storage in a database file.
15       * <p>
16       *     <em>Please note</em>: There is minimal help here.  You
17       *     need to keep track of which values will be encoded here.
18       *     If you are missing any, this method won't complain.
19       * </p>
20       */
21      public static String serializeHelper(Object... values) {
22          StringBuilder sb = new StringBuilder();
23          for (int i = 0; i < values.length-1; i++) {
24              String value = values[i] == null ? null : values[i].toString();
25              sb.append(StringUtils.encode(value)).append("|");
26          }
27          // append the last value with no pipe symbol afterwards
28          String lastValue = values[values.length - 1] == null ? null : values[values.length - 1].toString();
29          sb.append(StringUtils.encode(lastValue));
30          return sb.toString();
31      }
32  
33      /**
34       * Splits up a string based on a pipe character.  See {@link #tokenizer(String, char)}
35       *
36       * @param serializedText the string we are splitting into tokens
37       */
38      public static List<String> deserializeHelper(String serializedText) {
39          return tokenizer(serializedText, '|').stream().map(StringUtils::decode).toList();
40      }
41  
42      /**
43       * Splits up a string into tokens.
44       * @param serializedText the string we are splitting up
45       * @param delimiter the character acting as a boundary between sections
46       * @return a list of strings.  If the delimiter is not found, we will just return the whole string
47       */
48      public static List<String> tokenizer(String serializedText, char delimiter) {
49          final var resultList = new ArrayList<String>();
50          var currentPlace = 0;
51          int maxTokens = 200;
52          for(int i = 0; i <= maxTokens; i++) {
53              final var nextPipeSymbolIndex = serializedText.indexOf(delimiter, currentPlace);
54              if (nextPipeSymbolIndex == -1) {
55                  // if we don't see any pipe symbols ahead, grab the rest of the text from our current place
56                  resultList.add(serializedText.substring(currentPlace));
57                  break;
58              }
59              resultList.add(serializedText.substring(currentPlace, nextPipeSymbolIndex));
60              currentPlace = nextPipeSymbolIndex + 1;
61          }
62  
63          return resultList;
64      }
