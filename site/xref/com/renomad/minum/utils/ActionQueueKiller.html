7   /**
8    * This class exists to properly kill off multiple action queues
9    */
10  public final class ActionQueueKiller {
11  
12      private final Context context;
13      private final Constants constants;
14  
15      public ActionQueueKiller(Context context) {
16          this.context = context;
17          this.constants = context.getConstants();
18      }
19  
20      /**
21       * Systematically stops and kills all the action queues that have been
22       * instantiated in this call tree.
23       */
24      public void killAllQueues() {
25          if (constants.LOG_LEVELS.contains(LoggingLevel.DEBUG)) System.out.println(TimeUtils.getTimestampIsoInstant() + " Killing all queue threads. ");
26          for (ActionQueue aq : context.getActionQueueList()) {
27              aq.stop();
28              if (constants.LOG_LEVELS.contains(LoggingLevel.DEBUG)) System.out.println(TimeUtils.getTimestampIsoInstant() + " killing " + aq.getQueueThread());
29              if (aq.getQueueThread() != null) {
30                  aq.getQueueThread().interrupt();
31              }
32              aq.getQueue().clear();
33  
34          }
35          context.getActionQueueList().clear();
36      }
37  }
