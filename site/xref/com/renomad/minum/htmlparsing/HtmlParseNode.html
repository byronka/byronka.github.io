1   package com.renomad.minum.htmlparsing;
2   
3   import java.util.ArrayList;
4   import java.util.List;
5   import java.util.Map;
6   
7   /**
8    * These nodes represent the types of things we may encounter when parsing an HTML string, which
9    * for our purposes is just the types in {@link ParseNodeType}.  Depending on the type,
10   * different values will be filled out, but by coalescing them into this one class, it simplifies
11   * our code.
12   * See <a href="https://www.w3.org/TR/2011/WD-html-markup-20110113/syntax.html#syntax-elements">W3.org Elements</a>
13   */
14  public record HtmlParseNode(ParseNodeType type,
15                              TagInfo tagInfo,
16                              List<HtmlParseNode> innerContent,
17                              String textContent) {
18  
19      public static final HtmlParseNode EMPTY = new HtmlParseNode(ParseNodeType.ELEMENT, TagInfo.EMPTY, List.of(), "EMPTY HTMLPARSENODE");
20  
21      /**
22       * This method traverses the tree from this node downwards,
23       * adding the text content as it goes.
24       */
25      public List<String> print() {
26          var myList = new ArrayList<String>();
27          recursiveTreeWalk(myList);
28          return myList;
29      }
30  
31      private void recursiveTreeWalk(ArrayList<String> myList) {
32          for (var hpn : innerContent) {
33              hpn.recursiveTreeWalk(myList);
34          }
35          if (textContent != null && ! textContent.isBlank()) {
36              myList.add(textContent);
37          }
38      }
39  
40      /**
41       * This searches for a single node in the HTML.  If it
42       * finds more than one match, it will throw an exception.
43       * If none are found, HtmlParseNode.EMPTY is returned.
44       */
45      public List<HtmlParseNode> search(TagName tagName, Map<String, String> attributes) {
46          var myList = new ArrayList<HtmlParseNode>();
47          recursiveTreeWalkSearch(myList, tagName, attributes);
48          return myList;
49      }
50  
51      private void recursiveTreeWalkSearch(ArrayList<HtmlParseNode> myList, TagName tagName, Map<String, String> attributes) {
52          if (this.tagInfo().tagName().equals(tagName) && this.tagInfo().attributes().entrySet().containsAll(attributes.entrySet())) {
53              myList.add(this);
54          }
55          for (var htmlParseNode : innerContent) {
56              htmlParseNode.recursiveTreeWalkSearch(myList, tagName, attributes);
57          }
58      }
59  
60      /**
61       * If the element you're looking at has just one inner
62       * content item, and it's a CHARACTERS element, return it.
63       */
64      public String innerText() {
65         if (innerContent.size() == 1 && innerContent.get(0).type == ParseNodeType.CHARACTERS) {
66             return innerContent.get(0).textContent;
67         } else {
68             return "";
69         }
70      }
71  
72  }
