4   import java.util.ArrayList;
5   import java.util.List;
6   import java.util.Map;
7   
8   /**
9    * Allows rendering of templates
10   *
11   * <p>
12   * In order to perform speedy template rendering, it is necessary
13   * to get the template converted into this class.  The way to do
14   * that is as follows.
15   * </p>
16   * <p>
17   * First, you will be started with some suitable template. The values
18   * to be substituted are surrounded by double brackets.  Here's an example:
19   * </p>
20   * <pre>
21   * Hello, my name is {{name}}
22   * </pre>
23   * <p>
24   * Then, feed that string into {@link #buildProcessor}, like
25   * this:
26   * </p>
27   * <pre>
28   * <code>
29   * {@code String input = "Hello, my name is {{name}}"
30   * var helloProcessor = TemplateProcessor.buildProcessor(input);}
31   * </code>
32   * </pre>
33   * <p>
34   * Now that you have built a template processor, hold onto it! The generation of
35   * the template processor is the costly activity.  After that, you can use it
36   * efficiently, by feeding it a {@link Map} of key names to desired values. For
37   * our example, maybe we want <em>name</em> to be replaced with <em>Susanne</em>.
38   * In that case, we would do this:
39   * </p>
40   * <pre>
41   * <code>
42   * {@code var myMap = Map.of("name", "Susanne");
43   * String fullyRenderedString = helloProcessor.renderTemplate(myMap);}
44   * </code>
45   * </pre>
46   * <p>
47   *     The result is: Hello, my name is Susanne
48   * </p>
49   */
50  public final class TemplateProcessor {
51  
52      private final List<TemplateSection> templateSections;
53  
54      /**
55       * Instantiate a new object with a list of {@link TemplateSection}.
56       */
57      private TemplateProcessor(List<TemplateSection> templateSections) {
58          this.templateSections = templateSections;
59      }
60  
61      /**
62       * Given a map of key names -> value, render a template.
63       */
64      public String renderTemplate(Map<String, String> myMap) {
65          StringBuilder sb = new StringBuilder();
66          for (TemplateSection templateSection : templateSections) {
67              sb.append(templateSection.render(myMap));
68          }
69          return sb.toString();
70      }
71  
72      /**
73       * Builds a {@link TemplateProcessor} from a string
74       * containing a proper template.  Templated values
75       * are surrounded by double-curly-braces, i.e. {{foo}} or {{ foo }}
76       */
77      public static TemplateProcessor buildProcessor(String template) {
78          var tSections = new ArrayList<TemplateSection>();
79          StringBuilder builder = new StringBuilder();
80          // this flag is to help us understand whether we are currently reading the
81          // name of a template literal.
82          // e.g. in the case of hello {{ name }}, "name" is the literal.
83          boolean isInsideTemplateKeyLiteral = false;
84          for (int i = 0; i < template.length(); i++) {
85  
86              char charAtCursor = template.charAt(i);
87  
88              if (charAtCursor == '{' && (i + 1) < template.length() && template.charAt(i + 1) == '{') {
89                  // we were outside, now we're inside.
90                  isInsideTemplateKeyLiteral = true;
91                  i += 1;
92                  if (builder.length() > 0) {
93                      tSections.add(new TemplateSection(null, builder.toString()));
94                      builder = new StringBuilder();
95                  }
96                  continue;
97              }
98  
99              if (charAtCursor == '}' && (i + 1) < template.length() && template.charAt(i + 1) == '}' && isInsideTemplateKeyLiteral) {
100                 // we were inside, now we're outside.
101                 isInsideTemplateKeyLiteral = false;
102                 i += 1;
103                 if (builder.length() > 0) {
104                     tSections.add(new TemplateSection(builder.toString().trim(), null));
105                     builder = new StringBuilder();
106                 }
107                 continue;
108             }
109 
110             builder.append(charAtCursor);
111 
112             /*
113              if we're at the end of the template, it's our last chance to
114              add a substring (we can't be adding to a key, since if we're
115              at the end, and it's not a closing brace, it's a malformed
116              template.
117              */
118             if (i == template.length() - 1) {
119                 if (isInsideTemplateKeyLiteral) {
120                     // if we're exiting this string while inside a template literal, then
121                     // we're reading a corrupted input, and we should make that clear
122                     // to our caller.
123                     String templateSample = template.length() > 10 ? template.substring(0, 10) : template;
124                     throw new TemplateParseException("parsing failed for string starting with " + templateSample);
125                 }
126                 tSections.add(new TemplateSection(null, builder.toString()));
127             }
128         }
129 
130         return new TemplateProcessor(tSections);
131     }
132 }
