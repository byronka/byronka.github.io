3   /**
4    * Represents one item in the list that will eventually be cooked
5    * up into a fully-rendered string.  This record is the magic
6    * ingredient to an easy templating system. If it has a key,
7    * then this object will be getting replaced during final string rendering.  If it has a substring,
8    * then the substring gets concatenated unchanged when the final string
9    * is rendered.
10   */
11  class TemplateSection {
12  
13      public final String key;
14      public final int indent;
15      public final String staticData;
16      public final TemplateType templateType;
17      public final TemplateProcessor templateProcessor;
18  
19      /**
20       * @param indent the column number, measured from the left, of the first character of this template key.  This is used
21       *               to indent the subsequent lines of text when there are newlines in the replacement
22       *               text. For example, if the indent is 5, and the value is "a", then it should indent like this:
23       *               <br>
24       *               <pre>{@code
25       *               12345
26       *                   a
27       *               }</pre>
28       * @param key the name of the key, e.g. "name", or "color"
29       * @param subString the template content around the keys.  For example, in the text
30       *                  of "my favorite color is {{ color }} and I like it",
31       *                  it would generate three template sections - "my favorite color is" would be
32       *                  the first subString, then a key of "color", then a third subString of "and I like it"
33       */
34      public TemplateSection(String key, String subString, TemplateProcessor templateProcessor, TemplateType templateType, int indent) {
35          this.key = key;
36          this.staticData = subString;
37          this.templateProcessor = templateProcessor;
38          this.templateType = templateType;
39          this.indent = indent;
40      }
41  
42  }
