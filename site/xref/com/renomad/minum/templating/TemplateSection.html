3   import java.util.Map;

5   import static com.renomad.minum.utils.Invariants.mustBeTrue;

7   /**
8    * Represents one item in the list that will eventually be cooked
9    * up into a fully-rendered string.  This record is the magic
10   * ingredient to an easy templating system. If it has a key,
11   * then this object will be getting replaced during final string rendering.  If it has a substring,
12   * then the substring gets concatenated unchanged when the final string
13   * is rendered.
14   */
15  public record TemplateSection(String key, String subString) {
16  
17      public String render(Map<String, String> myMap) {
18          mustBeTrue(subString != null || key != null, "Either the key or substring must exist");
19          if (subString != null) {
20              mustBeTrue(key == null, "If this object has a substring, then it must not have a key");
21              return subString;
22          } else {
23              var value = myMap.get(key);
24              if (value == null) throw new TemplateRenderException("Missing a value for key {"+key+"}");
25              return value;
26          }
27      }
28  
29  }
