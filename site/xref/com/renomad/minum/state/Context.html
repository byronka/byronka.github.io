13  /**
14   * Holds important system-wide data and methods.
15   * <p>
16   *     Creating an instance of this and passing it
17   *     around lets us reduce some boilerplate code,
18   *     but more importantly, it lets us lower the scope.
19   * </p>
20   */
21  public final class Context {
22  
23      public static final Context EMPTY = new Context(null, null);
24      private ILogger logger;
25      private final ExecutorService executorService;
26      private final Constants constants;
27      private FullSystem fullSystem;
28      private final ActionQueueState actionQueueState;
29  
30      public Context(ExecutorService executorService, Constants constants) {
31          this.executorService = executorService;
32          this.constants = constants;
33          actionQueueState = new ActionQueueState();
34      }
35  
36      public void setLogger(ILogger logger) {
37          this.logger = logger;
38      }
39  
40      public ILogger getLogger() {
41          return logger;
42      }
43  
44      public ExecutorService getExecutorService() {
45          return executorService;
46      }
47  
48      public Constants getConstants() {
49          return constants;
50      }
51  
52      public void setFullSystem(FullSystem fullSystem) {
53          this.fullSystem = fullSystem;
54      }
55  
56      public FullSystem getFullSystem() {
57          return fullSystem;
58      }
59  
60      public ActionQueueState getActionQueueState() {
61          return actionQueueState;
62      }
63  
64      /**
65       * This is a helper method to instantiate a {@link Db} class,
66       * avoiding the need for a user to provide the root database
67       * directory and the context.
68       * <p>
69       * Since this is a generic method, a bit of care is required when
70       * calling.  Try to use a pattern like this:
71       * <pre>
72       * {@code Db<Photograph> photoDb = context.getDb("photos");}
73       * </pre>
74       * @param name the name of this data.  Note that this will be used
75       *             as the directory for the data, so use characters your
76       *             operating system would allow.
77       * @param instance an instance of the {@link DbData} data, preferably
78       *                 following a null-object pattern.  For example,
79       *                 Photograph.EMPTY.  This is used in the Db code
80       *                 to deserialize the data when reading.
81       */
82      public <T extends DbData<?>> Db<T> getDb(String name, T instance) {
83          return new Db<>(Path.of(constants.dbDirectory, name), this, instance);
84      }
