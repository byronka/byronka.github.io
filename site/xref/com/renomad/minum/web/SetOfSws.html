13  /**
14   * This is a data structure of the live set of {@link ISocketWrapper}
15   * in our system.  It exists so that we can keep tabs on how many
16   * open sockets we have, and can then find them all in one place
17   * when we need to kill them at server shutdown.
18   * @param nameOfSet This parameter is used to distinguish different servers'
19   *        list of sockets (e.g.
20   *        the server for 80 vs 443)
21   */
22  record SetOfSws(
23          ConcurrentSet<ISocketWrapper> data,
24          ILogger logger,
25          String nameOfSet) {
26  
27      void add(ISocketWrapper sw) {
28          data().add(sw);
29          int size = data().size();
30          logger.logTrace(() -> nameOfSet + " added " + sw + " to SetOfSws. size: " + size);
31      }
32  
33      void remove(ISocketWrapper sw) {
34          data().remove(sw);
35          int size = data().size();
36          logger.logTrace(() -> nameOfSet +" removed " + sw + " from SetOfSws. size: " + size);
37      }
38  
39      /**
40       * This is a program used during testing so we can find the server
41       * socket that corresponds to a particular client socket.
42       * <p>
43       * Due to the circumstances of the TCP handshake, there's a bit of
44       * time where the server might not have finished initialization,
45       * and been put into the list of current server sockets.
46       * <p>
47       * For that reason, if we come in here and don't find it initially, we'll
48       * sleep and then try again, up to three times.
49       */
50      ISocketWrapper getSocketWrapperByRemoteAddr(String address, int port) {
51          int maxLoops = 3;
52          for (int loopCount = 0; loopCount < maxLoops; loopCount++ ) {
53              List<ISocketWrapper> servers = data()
54                      .asStream()
55                      .filter(x -> x.getRemoteAddrWithPort().equals(new InetSocketAddress(address, port)))
56                      .toList();
57              mustBeFalse(servers.size() > 1, "Too many sockets found with that address");
58              if (servers.size() == 1) {
59                  return servers.get(0);
60              }
61  
62              // if we got here, we didn't find a server in the list - probably because the TCP
63              // initialization has not completed.  Retry after a bit.  The TCP process is dependent on
64              // a whole lot of variables outside our control - downed lines, slow routers, etc.
65              //
66              // on the other hand, this code should probably only be called in testing, so maybe fewer
67              // out-of-bounds problems?
68              int finalLoopCount = loopCount;
69              logger.logDebug(() -> String.format("no server found, sleeping on it... (attempt %d)", finalLoopCount + 1));
70              MyThread.sleep(10);
71          }
72          throw new RuntimeException("No socket found with that address");
73      }
74  
75      void stopAllServers() throws IOException {
76          for(var s : data()) {
77              s.close();
78          }
79      }
80  }
