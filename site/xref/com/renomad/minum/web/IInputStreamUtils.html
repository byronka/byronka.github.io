3   import java.io.IOException;
4   import java.io.InputStream;
5   
6   /**
7    * An interface for the {@link InputStreamUtils} implementation.
8    * Solely created to provide better testing access
9    */
10  interface IInputStreamUtils {
11  
12      /**
13       * Reads a line of text, stopping when reading a newline.
14       * <br>
15       * Skips over carriage returns, so we read a HTTP_CRLF properly.
16       * <br>
17       * <em>Note</em>: This is not a general-purpose line reader.  It is custom-designed
18       * for the inner workings of the Minum web server.  If you have need of
19       * a typical {@link InputStream} line reader, it is probably better to
20       * use {@link java.io.BufferedReader}
21       * @return     A String containing the contents of the line, not including
22       *             any line-termination characters, or null if the end of the
23       *             stream has been reached without reading any characters
24       *
25       * @throws     IOException  If an I/O error occurs
26       */
27      String readLine(InputStream inputStream) throws IOException;
28  
29      /**
30       * Reads "lengthToRead" bytes from the input stream
31       */
32      byte[] read(int lengthToRead, InputStream inputStream);
33  }
