3   import java.util.Arrays;
4   import java.util.regex.Matcher;
5   import java.util.regex.Pattern;
6   
7   import static com.renomad.minum.utils.Invariants.mustBeTrue;

9   /**
10   * This class represents the text that is sent back in a {@link Response}
11   */
12  public record StatusLine(StatusCode status, HttpVersion version, String rawValue) {
13  
14      static StatusLine EMPTY = new StatusLine(StatusCode.NULL, HttpVersion.NONE, "");
15  
16      /**
17       * This is the regex used to analyze a status line sent by the server and
18       * read by the client.  Servers will send messages like: "HTTP/1.1 200 OK" or "HTTP/1.1 500 Internal Server Error"
19       */
20      static final String statusLinePattern = "^HTTP/(1.1|1.0) (\\d{3}) (.*)$";
21      static final Pattern statusLineRegex = Pattern.compile(statusLinePattern);
22  
23      /**
24       * See see <a href="https://developer.mozilla.org/en-US/docs/Web/HTTP/Status">Status Codes</a>
25       */
26      public enum StatusCode{
27  
28          /* Successful responses (200 – 299) */
29  
30          _200_OK(200, "OK"),
31          _201_CREATED(201, "CREATED"),
32          _202_ACCEPTED(202, "ACCEPTED"),
33          _204_NO_CONTENT(204, "NO CONTENT"),
34  
35          /* Redirection messages (300 – 399) */
36  
37          _301_MOVED_PERMANENTLY(301, "MOVED PERMANENTLY"),
38          _302_FOUND(302, "FOUND"),
39  
40          /**
41           * Used a lot after receiving a post response.  The pattern is to
42           * receive the post, then redirect to a new page. See <a href="https://en.wikipedia.org/wiki/Post/Redirect/Get">...</a>
43           */
44          _303_SEE_OTHER(303, "SEE OTHER"),
45          _304_NOT_MODIFIED(304, "NOT MODIFIED"),
46          _307_TEMPORARY_REDIRECT(307, "TEMPORARY REDIRECT"),
47          _308_PERMANENT_REDIRECT(308, "PERMANENT REDIRECT"),
48  
49          /* Client error responses (400 – 499) */
50  
51          _400_BAD_REQUEST(400, "BAD REQUEST"),
52          _401_UNAUTHORIZED(401, "UNAUTHORIZED"),
53          _403_FORBIDDEN(403, "FORBIDDEN"),
54          _404_NOT_FOUND(404, "NOT FOUND"),
55          _405_METHOD_NOT_ALLOWED(405, "METHOD NOT ALLOWED"),
56          _406_NOT_ACCEPTABLE(406, "NOT ACCEPTABLE"),
57          _408_REQUEST_TIMEOUT(408, "REQUEST TIMEOUT"),
58          _409_CONFLICT(409, "CONFLICT"),
59          _410_GONE(410, "GONE"),
60          _411_LENGTH_REQUIRED(411, "LENGTH REQUIRED"),
61          _413_PAYLOAD_TOO_LARGE(413, "PAYLOAD TOO LARGE"),
62          _414_URI_TOO_LONG(414, "URI TOO LONG"),
63          _415_UNSUPPORTED_MEDIA_TYPE(415, "UNSUPPORTED MEDIA TYPE"),
64          _426_UPGRADE_REQUIRED(426, "UPGRADE REQUIRED"),
65          _429_TOO_MANY_REQUESTS(429, "TOO MANY REQUESTS"),
66          _431_REQUEST_HEADER_FIELDS_TOO_LARGE(431, "REQUEST HEADER FIELDS TOO LARGE"),
67  
68          /* Server error responses (500 – 599) */
69  
70          _500_INTERNAL_SERVER_ERROR(500, "INTERNAL SERVER ERROR"),
71          _501_NOT_IMPLEMENTED(501, "NOT IMPLEMENTED"),
72          _502_BAD_GATEWAY(502, "BAD GATEWAY"),
73          _503_SERVICE_UNAVAILABLE(503, "SERVICE UNAVAILABLE"),
74          _504_GATEWAY_TIMEOUT(504, "GATEWAY TIMEOUT"),
75          _505_HTTP_VERSION_NOT_SUPPORTED(505, "HTTP VERSION NOT SUPPORTED"),
76  
77          /**
78           * The null object, meant to represent "no status code"
79           */
80          NULL(0, "NULL OBJECT")
81          ;
82  
83          public final int code;
84          public final String shortDescription;
85  
86          StatusCode(int code, String shortDescription) {
87              this.code = code;
88              this.shortDescription = shortDescription;
89          }
90  
91          static StatusCode findByCode(int code) {
92              return Arrays.stream(StatusCode.values())
93                      .filter(x -> x.code == code)
94                      .findFirst()
95                      .orElseThrow();
96          }
97      }
98  
99      /**
100      * Parses a string value of a status line from an HTTP
101      * server.  If the input value is null or empty, we'll
102      * return a {@link StatusLine} with null-object values
103      */
104     public static StatusLine extractStatusLine(String value) {
105         if (value == null || value.isBlank()) {
106             return StatusLine.EMPTY;
107         }
108         Matcher mr = StatusLine.statusLineRegex.matcher(value);
109         mustBeTrue(mr.matches(), String.format("%s must match the statusLinePattern: %s", value, statusLinePattern));
110         String version = mr.group(1);
111         HttpVersion httpVersion = switch (version) {
112             case "1.1" -> HttpVersion.ONE_DOT_ONE;
113             case "1.0" -> HttpVersion.ONE_DOT_ZERO;
114             default -> throw new RuntimeException(String.format("HTTP version was not an acceptable value. Given: %s", version));
115         };
116         StatusCode status = StatusCode.findByCode(Integer.parseInt(mr.group(2)));
117 
118         return new StatusLine(status, httpVersion, value);
119     }
120 }
