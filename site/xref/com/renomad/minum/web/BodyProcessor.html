19  /**
20   * This code is responsible for extracting the {@link Body} from
21   * an HTTP request.
22   */
23  final class BodyProcessor {
24  
25      private final ILogger logger;
26      private final InputStreamUtils inputStreamUtils;
27      private final Context context;

29      /**
30       * When parsing fails, we would like to send the raw text
31       * back to the user so the development team can determine
32       * why parsing went awry.  But, when we are sent a huge file,
33       * we would rather not include all that data in the logs.
34       * So we will cap out at this value.
35       */
36      static final int MAX_SIZE_DATA_RETURNED_IN_EXCEPTION = 1024;
37  
38      BodyProcessor(Context context) {
39          this.context = context;
40          this.logger = context.getLogger();
41          this.inputStreamUtils = context.getInputStreamUtils();
42      }
43  
44      /**
45       * read the body if one exists
46       * <br>
47       * There are really only two ways to read the body.
48       * 1. the client tells us how many bytes to read
49       * 2. the client uses "transfer-encoding: chunked"
50       * <br>
51       * In either case, it is absolutely critical that the client gives us
52       * a way to know ahead of time how many bytes to read, so we (the server)
53       * can stop reading at precisely the right point.  There's simply no
54       * other way to reasonably do this.
55       */
56      Body extractData(InputStream is, Headers h) throws IOException {
57          final var contentType = h.contentType();
58  
59          byte[] bodyBytes = h.contentLength() > 0 ?
60                  inputStreamUtils.read(h.contentLength(), is) :
61                  inputStreamUtils.readChunkedEncoding(is);
62  
63          return extractBodyFromBytes(h.contentLength(), contentType, bodyBytes);
64      }

66      /**
67       * Handles the parsing of the body data for either form-urlencoded or
68       * multipart/form-data
69       * @param contentType a mime value which must be either application/x-www-form-urlencoded
70       *                    or multipart/form-data.  Anything else will cause a new Body to
71       *                    be created with the body bytes, unparsed.  There are a number of
72       *                    cases where this makes sense - if the user is sending us plain text,
73       *                    html, json, or css, we want to simply accept the data and not try to parse it.
74       * @param bodyBytes the full body of this HTTP message, as bytes.
75       */
76      Body extractBodyFromBytes(int contentLength, String contentType, byte[] bodyBytes) {
77          try {
78              if (contentLength > 0 && contentType.contains("application/x-www-form-urlencoded")) {
79                  return parseUrlEncodedForm(StringUtils.byteArrayToString(bodyBytes));
80              } else if (contentType.contains("multipart/form-data")) {
81                  String boundaryKey = "boundary=";
82                  int indexOfBoundaryKey = contentType.indexOf(boundaryKey);
83                  if (indexOfBoundaryKey > 0) {
84                      // grab all the text after the key
85                      String boundaryValue = contentType.substring(indexOfBoundaryKey + boundaryKey.length());
86                      return parseMultiform(bodyBytes, boundaryValue);
87                  }
88                  String parsingError = "Did not find a valid boundary value for the multipart input. Returning an empty map and the raw bytes for the body. Header was: " + contentType;
89                  logger.logDebug(() -> parsingError);
90                  return new Body(Map.of(), bodyBytes, Map.of());
91              } else {
92                  logger.logDebug(() -> "did not recognize a key-value pattern content-type, returning an empty map and the raw bytes for the body");
93                  return new Body(Map.of(), bodyBytes, Map.of());
94              }
95          } catch (Exception ex) {
96              logger.logDebug(() -> "Unable to parse this body. returning an empty map and the raw bytes for the body.  Exception message: " + ex.getMessage());
97              return new Body(Map.of(), bodyBytes, Map.of());
98          }
99      }

102     /**
103      * Parse data formatted by application/x-www-form-urlencoded
104      * See <a href="https://developer.mozilla.org/en-US/docs/Web/HTTP/Methods/POST">...</a>
105      * <p>
106      * See here for the encoding: <a href="https://developer.mozilla.org/en-US/docs/Glossary/percent-encoding">...</a>
107      * <p>
108      * for example, {@code valuea=3&valueb=this+is+something}
109      */
110     Body parseUrlEncodedForm(String input) {
111         if (input.isEmpty()) return Body.EMPTY;
112         final var postedPairs = new HashMap<String, byte[]>();
113 
114         try {
115             final var splitByAmpersand = tokenizer(input, '&');
116 
117             for (final var s : splitByAmpersand) {
118                 final var pair = splitKeyAndValue(s);
119                 mustBeTrue(pair.length == 2, "Splitting on = should return 2 values.  Input was " + s);
120                 mustBeTrue(!pair[0].isBlank(), "The key must not be blank");
121                 final var value = StringUtils.decode(pair[1]);
122                 final var convertedValue = value == null ? "".getBytes(StandardCharsets.UTF_8) : value.getBytes(StandardCharsets.UTF_8);
123                 final var result = postedPairs.put(pair[0], convertedValue);
124 
125                 if (result != null) {
126                     throw new InvariantException(pair[0] + " was duplicated in the post body - had values of " + StringUtils.byteArrayToString(result) + " and " + pair[1]);
127                 }
128             }
129         } catch (Exception ex) {
130             String dataToReturn = input;
131             if (input.length() > MAX_SIZE_DATA_RETURNED_IN_EXCEPTION) {
132                 dataToReturn = input.substring(0, MAX_SIZE_DATA_RETURNED_IN_EXCEPTION) + " ... (remainder of data trimmed)";
133             }
134             logger.logDebug(() -> "Unable to parse this body. returning an empty map and the raw bytes for the body.  Exception message: " + ex.getMessage());
135             return new Body(Map.of(), dataToReturn.getBytes(StandardCharsets.UTF_8), Map.of());
136         }
137         return new Body(postedPairs, input.getBytes(StandardCharsets.UTF_8), Map.of());
138     }

140     /**
141      * This splits a key from its value, following the HTTP pattern
142      * of "key=value". (that is, a key string, concatenated to an "equals"
143      * character, concatenated to the value, with no spaces [and the key
144      * and value are URL-encoded])
145      * @param formInputValue a string like "key=value"
146      */
147     private String[] splitKeyAndValue(String formInputValue) {
148         final var locationOfEqual = formInputValue.indexOf("=");
149         return new String[] {
150                 formInputValue.substring(0, locationOfEqual),
151                 formInputValue.substring(locationOfEqual+1)};
152     }

154     /**
155      * A regex used to extract the name value from the headers in multipart/form
156      * For example, in the following code, you can see that the name is "image_uploads"
157      * <pre>
158      * {@code
159      * --i_am_a_boundary
160      * Content-Type: text/plain
161      * Content-Disposition: form-data; name="text1"
162      *
163      * I am a value that is text
164      * --i_am_a_boundary
165      * Content-Type: application/octet-stream
166      * Content-Disposition: form-data; name="image_uploads"; filename="photo_preview.jpg"
167      * }
168      * </pre>
169      */
170     private static final Pattern multiformNameRegex = Pattern.compile("\\bname\\b=\"(?<namevalue>.*?)\"");
171 
172     /**
173      * Extract multipart/form data from a body.  See docs/http_protocol/returning_values_from_multipart_rfc_7578.txt
174      */
175     Body parseMultiform(byte[] body, String boundaryValue) {
176         // how to split this up? It's a mix of strings and bytes.
177         List<byte[]> partitions = split(body, "--" + boundaryValue);
178         // What we can bear in mind is that once we've read the headers, and gotten
179         // past the single blank line, *everything else* is pure data.
180         final var result = new HashMap<String, byte[]>();
181         final var partitionHeaders = new HashMap<String, Headers>();
182         for (var df : partitions) {
183             final var is = new ByteArrayInputStream(df);
184 
185             Headers headers;
186             try {
187                 headers = Headers.make(context, inputStreamUtils).extractHeaderInformation(is);
188             } catch (IOException e) {
189                 // This is an InputStream we built ourselves from local data.
190                 // there's no reason why it should fail.
191                 throw new WebServerException(e);
192             }
193 
194             List<String> cds = headers.valueByKey("Content-Disposition");
195             String contentDisposition = String.join(";", cds == null ? List.of("") : cds);
196 
197             Matcher matcher = multiformNameRegex.matcher(contentDisposition);
198 
199             if (matcher.find()) {
200                 String name = matcher.group("namevalue");
201                 // at this point our inputstream pointer is at the beginning of the
202                 // body data.  From here until the end it's pure data.
203                 byte[] data;
204                 try {
205                     data = inputStreamUtils.readUntilEOF(is);
206                 } catch (IOException e) {
207                     // This is an InputStream we built ourselves from local data.
208                     // there's no reason why it should fail.
209                     throw new WebServerException(e);
210                 }
211                 result.put(name, data);
212                 partitionHeaders.put(name, headers);
213             }
214             else {
215                 String returnedData;
216                 if (body.length > MAX_SIZE_DATA_RETURNED_IN_EXCEPTION) {
217                     returnedData = StringUtils.byteArrayToString(Arrays.copyOf(body, MAX_SIZE_DATA_RETURNED_IN_EXCEPTION)) + " ... (remainder of data trimmed)";
218                 } else {
219                     returnedData = StringUtils.byteArrayToString(body);
220                 }
221                 logger.logDebug(() -> "No name value found in the headers of a partition. Data: " + returnedData);
222                 return new Body(Map.of(), body, Map.of());
223             }
224 
225         }
226         return new Body(result, body, partitionHeaders);
227     }

229     /**
230      * Given a multipart-formatted data, return a list of byte arrays
231      * between the boundary values
232      */
233     List<byte[]> split(byte[] body, String boundaryValue) {
234         List<byte[]> result = new ArrayList<>();
235         List<Integer> indexesOfEndsOfBoundaries = new ArrayList<>();
236         byte[] boundaryValueBytes = boundaryValue.getBytes(StandardCharsets.UTF_8);
237         int indexIntoBoundaryValue = 0;
238         for (int i = 0; i < body.length; i++) {
239             if (body[i] == boundaryValueBytes[indexIntoBoundaryValue]) {
240                 if (indexIntoBoundaryValue == boundaryValueBytes.length - 1) {
241                     // have to add two to account for *either* CR+LF or two dashes.  Multipart
242                     // form is so complicated!
243                     indexesOfEndsOfBoundaries.add(i + 2);
244                     indexIntoBoundaryValue = 0;
245                 }
246                 indexIntoBoundaryValue += 1;
247             } else {
248                 indexIntoBoundaryValue = 0;
249             }
250         }
251         // now we know where the boundaries are in this hunk of data.  Partition time!
252         int indexInBody = 0;
253         for (int endOfBoundaryIndex : indexesOfEndsOfBoundaries) {
254             // the minus one plus two silliness is to make clear that we're calculating for two extra chars after the boundary
255             if (indexInBody != endOfBoundaryIndex-(boundaryValue.length() + 2 - 1)) {
256                 result.add(Arrays.copyOfRange(body, indexInBody, endOfBoundaryIndex-(boundaryValue.length() + 2 -1)));
257             }
258             indexInBody = endOfBoundaryIndex+1;
259         }
260         return result;
261     }

264     /**
265      * Splits up a string into tokens.
266      * @param serializedText the string we are splitting up
267      * @param delimiter the character acting as a boundary between sections
268      * @return a list of strings.  If the delimiter is not found, we will just return the whole string
269      */
270     private List<String> tokenizer(String serializedText, char delimiter) {
271         final var resultList = new ArrayList<String>();
272         var currentPlace = 0;
273         // when would we have a need to tokenize anything into more than MAX_TOKENIZER_PARTITIONS partitions?
274         for(int i = 0; i <= context.getConstants().MAX_TOKENIZER_PARTITIONS; i++) {
275             if (i == context.getConstants().MAX_TOKENIZER_PARTITIONS) throw new ForbiddenUseException("Request made for too many partitions in the tokenizer.  Current max: " + context.getConstants().MAX_TOKENIZER_PARTITIONS);
276             final var nextPipeSymbolIndex = serializedText.indexOf(delimiter, currentPlace);
277             if (nextPipeSymbolIndex == -1) {
278                 // if we don't see any pipe symbols ahead, grab the rest of the text from our current place
279                 resultList.add(serializedText.substring(currentPlace));
280                 break;
281             }
282             resultList.add(serializedText.substring(currentPlace, nextPipeSymbolIndex));
283             currentPlace = nextPipeSymbolIndex + 1;
284         }
285 
286         return resultList;
287     }
