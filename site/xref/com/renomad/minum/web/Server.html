18  /**
19   * The purpose here is to make it marginally easier to
20   * work with a ServerSocket.
21   * <p>
22   * First, instantiate this class using a running serverSocket
23   * Then, by running the start method, we gain access to
24   * the server's socket.  This way we can easily test / control
25   * the server side but also tie it in with an ExecutorService
26   * for controlling lots of server threads.
27   */
28  final class Server implements AutoCloseable {
29      private final ServerSocket serverSocket;
30      private final SetOfSws setOfSWs;
31      private final ILogger logger;
32      private final String serverName;
33      private final ITheBrig theBrig;
34      private final Constants constants;
35      private final UnderInvestigation underInvestigation;
36  
37      /**
38       * This is the future returned when we submitted the
39       * thread for the central server loop to the ExecutorService
40       */
41      private Future<?> centralLoopFuture;
42  
43      Server(ServerSocket ss, Context context, String serverName, ITheBrig theBrig) {
44          this.serverSocket = ss;
45          this.logger = context.getLogger();
46          this.constants = context.getConstants();
47          this.serverName = serverName;
48          this.theBrig = theBrig;
49          setOfSWs = new SetOfSws(new ConcurrentSet<>(), logger, serverName);
50          this.underInvestigation = new UnderInvestigation(constants);
51      }
52  
53      /**
54       * This is the infinite loop running inside the basic socket server code.  Every time we
55       * {@link ServerSocket#accept()} a new incoming socket connection, we attach our "end
56       * of the phone line" to the code that is "handler", which mainly handles it from there.
57       * @param handler the commonest handler will be found at {@link WebFramework#makePrimaryHttpHandler}
58       */
59      void start(ExecutorService es, ThrowingConsumer<ISocketWrapper, IOException> handler) {
60          ThrowingRunnable serverCode = buildMainServerLoop(es, handler);
61          Runnable t = ThrowingRunnable.throwingRunnableWrapper(serverCode, logger);
62          this.centralLoopFuture = es.submit(t);
63      }
64  
65      /**
66       * This code is the innermost loop of the server, waiting for incoming
67       * connections and then delegating their handling off to a handler.
68       * @param es The ExecutorService helping us with the threads
69       * @param handler the handler that will take charge immediately after
70       *                a client makes a connection.
71       */
72      private ThrowingRunnable buildMainServerLoop(ExecutorService es, ThrowingConsumer<ISocketWrapper, IOException> handler) {
73          ThrowingRunnable serverCode = () -> {
74              Thread.currentThread().setName("Main Server");
75              try {
76                  // yes, this infinite loop can only exit by an exception.  But this is
77                  // the beating heart of a server, and to the best of my current knowledge,
78                  // when a server socket is force-closed it's going to throw an exception, and
79                  // that's just part of its life cycle
80                  //noinspection InfiniteLoopStatement
81                  while (true) {
82                      logger.logTrace(() -> serverName + " waiting to accept connection");
83                      Socket freshSocket = serverSocket.accept();
84                      ISocketWrapper sw = new SocketWrapper(freshSocket, this, logger, constants.SOCKET_TIMEOUT_MILLIS);
85                      logger.logTrace(() -> String.format("client connected from %s", sw.getRemoteAddrWithPort()));
86                      setOfSWs.add(sw);
87                      if (handler != null) {
88                          ThrowingRunnable innerServerCode = buildExceptionHandlingInnerCore(handler, sw);
89                          es.submit(ThrowingRunnable.throwingRunnableWrapper(innerServerCode, logger));
90                      }
91                  }
92              } catch (IOException ex) {
93                  if (!(ex.getMessage().contains("Socket closed") || ex.getMessage().contains("Socket is closed"))) {
94                      logger.logAsyncError(() -> StacktraceUtils.stackTraceToString(ex));
95                  }
96              }
97          };
98          return serverCode;
99      }

101     /**
102      * By volume of code, this method is primarily focused on handling the kinds
103      * of exceptional situations that can arise from handling the HTTP communication
104      * with a client.
105      * <p>
106      *     Note that this method *builds* a function but does not run it.
107      * </p>
108      *
109      */
110     ThrowingRunnable buildExceptionHandlingInnerCore(ThrowingConsumer<ISocketWrapper, IOException> handler, ISocketWrapper sw) {
111         ThrowingRunnable innerServerCode = () -> {
112             Thread.currentThread().setName("SocketWrapper thread for " + sw.getRemoteAddr());
113             try {
114                 handler.accept(sw);
115             } catch (SocketException | SocketTimeoutException ex) {
116                  /*
117                  if we close the application on the server side, there's a good
118                  likelihood a SocketException will come bubbling through here.
119                  NOTE:
120                    it seems that Socket closed is what we get when the client closes the connection in non-SSL, and conversely,
121                    if we are operating in secure (i.e. SSL/TLS) mode, we get "an established connection..."
122                  */
123                 if (ex.getMessage().equals("Read timed out")) {
124                     logger.logTrace(() -> "Read timed out - remote address: " + sw.getRemoteAddrWithPort());
125                 } else {
126                     logger.logDebug(() -> ex.getMessage() + " - remote address: " + sw.getRemoteAddrWithPort());
127                 }
128 
129             } catch (ForbiddenUseException ex) {
130                 logger.logDebug(ex::getMessage);
131             } catch (IOException ex) {
132                 logger.logDebug(() -> ex.getMessage() + " (at Server.start)");
133                 String suspiciousClues = underInvestigation.isClientLookingForVulnerabilities(ex.getMessage());
134 
135                 if (suspiciousClues.length() > 0 && theBrig != null) {
136                     logger.logDebug(() -> sw.getRemoteAddr() + " is looking for vulnerabilities, for this: " + suspiciousClues);
137                     theBrig.sendToJail(sw.getRemoteAddr() + "_vuln_seeking", constants.VULN_SEEKING_JAIL_DURATION);
138                 }
139             }
140         };
141         return innerServerCode;
142     }
143 
144     public void close() throws IOException {
145         // close all the running sockets
146         setOfSWs.stopAllServers();
147         logger.logTrace(() -> "close called on " + this);
148         // close the primary server socket
149         serverSocket.close();
150     }

152     /**
153      * Get the string version of the address of this
154      * server.  See {@link InetAddress#getHostAddress()}
155      */
156     String getHost() {
157         return serverSocket.getInetAddress().getHostAddress();
158     }
159 
160     /**
161      * See {@link ServerSocket#getLocalPort()}
162      */
163     int getPort() {
164         return serverSocket.getLocalPort();
165     }

167     /**
168      * This is a helper method to find the server SocketWrapper
169      * connected to a client SocketWrapper.
170      */
171     ISocketWrapper getServer(ISocketWrapper sw) {
172         return setOfSWs.getSocketWrapperByRemoteAddr(sw.getLocalAddr(), sw.getLocalPort());
173     }
174 
175     /**
176      * When we first create a SocketWrapper in Server, we provide it
177      * a reference back to this object, so that it can call
178      * this command.  This class maintains a list of open
179      * sockets in setOfSWs, and allows generated SocketWrappers
180      * to deregister themselves from this list by using this method.
181      */
182     void removeMyRecord(ISocketWrapper socketWrapper) {
183         setOfSWs.remove(socketWrapper);
184     }
