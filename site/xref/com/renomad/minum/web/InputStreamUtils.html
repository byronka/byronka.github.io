6   import java.io.ByteArrayOutputStream;
7   import java.io.IOException;
8   import java.io.InputStream;
9   import java.nio.charset.StandardCharsets;
10  
11  /**
12   * Handy helpful utilities for working with input streams.
13   */
14  public final class InputStreamUtils implements IInputStreamUtils {
15  
16      public static final int MAX_READ_LINE_SIZE_BYTES = 1024;
17  
18      @Override
19      public String readLine(InputStream inputStream) throws IOException  {
20          final int NEWLINE_DECIMAL = 10;
21          final int CARRIAGE_RETURN_DECIMAL = 13;
22  
23          final var result = new ByteArrayOutputStream(MAX_READ_LINE_SIZE_BYTES / 3);
24          for (int i = 0;; i++) {
25              if (i >= MAX_READ_LINE_SIZE_BYTES) {
26                  inputStream.close();
27                  throw new ForbiddenUseException("client sent more bytes than allowed for a single line.  max: " + MAX_READ_LINE_SIZE_BYTES);
28              }
29              int a = inputStream.read();
30              if (a == -1) return result.toString(StandardCharsets.UTF_8);
31              if (a == CARRIAGE_RETURN_DECIMAL) continue;
32              if (a == NEWLINE_DECIMAL) break;
33              result.write(a);
34          }
35          return result.toString(StandardCharsets.UTF_8);
36      }
37  
38      @Override
39      public byte[] read(int lengthToRead, InputStream inputStream) {
40          final int typicalBufferSize = 1024 * 8;
41          byte[] buf = new byte[Math.min(lengthToRead, typicalBufferSize)]; // 8k buffer is my understanding of a decent size.  Fast, doesn't waste too much space.
42          byte[] data;
43          ByteArrayOutputStream baos = new ByteArrayOutputStream();
44          int read;
45          int totalRead = 0;
46          try {
47              while ((read = inputStream.read(buf)) >= 0) {
48                  totalRead += read;
49                  if (totalRead < lengthToRead) {
50                      // if we haven't gotten everything we wanted, write this to the output and loop again
51                      baos.write(buf, 0, read);
52                  } else {
53                      baos.write(buf, 0, read - (totalRead - lengthToRead));
54                      break;
55                  }
56              }
57          } catch (IOException ex) {
58              throw new UtilsException(ex);
59          }
60          data = baos.toByteArray();
61  
62          if (data.length != lengthToRead) {
63              String message = String.format("length of bytes read (%d) must be what we expected (%d)", data.length, lengthToRead);
64              throw new ForbiddenUseException(message);
65          }
66          return data;
67      }
68  }
