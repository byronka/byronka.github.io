30      /**
31       * Read from the socket until it returns an EOF indicator (that is, -1)
32       * Note: this *will block* until it gets to that EOF.
33       */
34      public byte[] readUntilEOF(InputStream inputStream) throws IOException {
35          final var result = new ArrayList<Byte>();
36          for (int i = 0; i <= (constants.MAX_READ_SIZE_BYTES + 1); i++) {
37              if (i == constants.MAX_READ_SIZE_BYTES) throw new ForbiddenUseException("client sent more bytes than allowed.  Current max: " + constants.MAX_READ_SIZE_BYTES);
38              int a = inputStream.read();
39              if (a == -1) {
40                  return byteListToArray(result);
41              }
42  
43              result.add((byte) a);
44          }
45          return byteListToArray(result);
46      }

48      /**
49       * reads following the algorithm for transfer-encoding: chunked.
50       * See <a href="https://en.wikipedia.org/wiki/Chunked_transfer_encoding">chunked transfer encoding</a>
51       */
52      public byte[] readChunkedEncoding(InputStream inputStream) throws IOException {
53          final var result = new ByteArrayOutputStream( );
54          int countRead = 0;
55          while (countRead <= (constants.MAX_READ_SIZE_BYTES + 1))  {
56              if (countRead == constants.MAX_READ_SIZE_BYTES) {
57                  logger.logDebug(() -> "client sent more bytes than allowed.  Current max: " + constants.MAX_READ_SIZE_BYTES);
58                  inputStream.close();
59                  return new byte[0];
60              }
61              String countToReadString = readLine(inputStream);
62              if (countToReadString == null) {
63                  return new byte[0];
64              }
65              mustNotBeNull(countToReadString);
66              int countToRead = Integer.parseInt(countToReadString, 16);
67  
68              result.write(read(countToRead, inputStream));
69              countRead += countToRead;
70  
71              readLine(inputStream);
72              if (countToRead == 0) {
73                  readLine(inputStream);
74                  break;
75              }
76  
77          }
78          return result.toByteArray();
79      }

81      /**
82       * Reads a line of text, stopping when reading a newline.
83       * Skips over carriage returns, so we read a HTTP_CRLF properly.
84       * <br>
85       * If the stream ends, return null
86       */
87      public String readLine(InputStream inputStream) throws IOException  {
88          final int NEWLINE_DECIMAL = 10;
89          final int CARRIAGE_RETURN_DECIMAL = 13;
90  
91          final var result = new ByteArrayOutputStream(constants.MAX_READ_LINE_SIZE_BYTES / 3);
92          for (int i = 0; i <= (constants.MAX_READ_LINE_SIZE_BYTES + 1); i++) {
93              if (i == constants.MAX_READ_LINE_SIZE_BYTES) {
94                  logger.logDebug(() -> String.format(
95                          "in readLine, client sent more bytes than allowed.  Current max: %d.  Contents: %s",
96                          constants.MAX_READ_LINE_SIZE_BYTES,
97                          result.toString(StandardCharsets.UTF_8)));
98                  inputStream.close();
99                  return "";
100             }
101             int a = inputStream.read();
102 
103             if (a == -1) return null;
104 
105             if (a == CARRIAGE_RETURN_DECIMAL) {
106                 continue;
107             }
108 
109             if (a == NEWLINE_DECIMAL) break;
110 
111             result.write(a);
112 
113         }
114         return result.toString(StandardCharsets.UTF_8);
115     }

118     /**
119      * Reads "lengthToRead" bytes from the input stream
120      */
121     public byte[] read(int lengthToRead, InputStream inputStream) throws IOException {
122         if (lengthToRead > constants.MAX_READ_SIZE_BYTES) {
123             throw new ForbiddenUseException("client requested to send more bytes than allowed.  Current max: " + constants.MAX_READ_SIZE_BYTES + " asked to receive: " + lengthToRead);
124         }
125         final int typicalBufferSize = 1024 * 8;
126         byte[] buf = new byte[Math.min(lengthToRead, typicalBufferSize)]; // 8k buffer is my understanding of a decent size.  Fast, doesn't waste too much space.
127         byte[] data;
128         ByteArrayOutputStream baos = new ByteArrayOutputStream();
129         int read;
130         int totalRead = 0;
131         while ((read = inputStream.read(buf)) >= 0) {
132             totalRead += read;
133             if (totalRead < lengthToRead) {
134                 // if we haven't gotten everything we wanted, write this to the output and loop again
135                 baos.write(buf, 0, read);
136             } else {
137                 baos.write(buf, 0, read - (totalRead - lengthToRead));
138                 break;
139             }
140         }
141         data = baos.toByteArray();
142 
143         mustBeTrue(data.length == lengthToRead, String.format("lengthToRead of bytes read (%d) must be what we expected (%d)", data.length, lengthToRead));
144         return data;
145     }
