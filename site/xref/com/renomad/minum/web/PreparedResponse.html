4   import java.util.Arrays;
5   import java.util.Objects;

7   /**
8    * This type is used to contain the results of modifications
9    * applied to the outgoing data just before sending
10   * @param statusLineAndHeaders string values separated by CRLF, per HTTP spec
11   * @param body the body of the message, as bytes.  May be compressed.
12   */
13  record PreparedResponse(String statusLineAndHeaders, byte[] body){
14      @Override
15      public boolean equals(Object o) {
16          if (this == o) return true;
17          if (o == null || getClass() != o.getClass()) return false;
18          PreparedResponse that = (PreparedResponse) o;
19          return Objects.equals(statusLineAndHeaders, that.statusLineAndHeaders) && Arrays.equals(body, that.body);
20      }
21  
22      @Override
23      public int hashCode() {
24          int result = Objects.hash(statusLineAndHeaders);
25          result = 31 * result + Arrays.hashCode(body);
26          return result;
27      }
28  
29      @Override
30      public String toString() {
31          return "PreparedResponse{" +
32                  "statusLineAndHeaders='" + statusLineAndHeaders + '\'' +
33                  ", body=" + Arrays.toString(body) +
34                  '}';
35      }
36  }
