1   package com.renomad.minum.web;

3   
4   import java.util.Arrays;
5   import java.util.Objects;
6   
7   /**
8    * This type is used to contain the results of modifications
9    * applied to the outgoing data just before sending
10   */
11  final class PreparedResponse {
12  
13      private final String statusLineAndHeaders;
14      private final byte[] body;
15  
16      /**
17       * Construct a {@link PreparedResponse}
18       * @param statusLineAndHeaders string values separated by CRLF, per HTTP spec
19       * @param body the body of the message, as bytes.  May be compressed.
20       */
21      PreparedResponse(String statusLineAndHeaders, byte[] body) {
22  
23          this.statusLineAndHeaders = statusLineAndHeaders;
24          this.body = body.clone();
25      }
26  
27      public String getStatusLineAndHeaders() {
28          return statusLineAndHeaders;
29      }
30  
31      public byte[] getBody() {
32          return body.clone();
33      }
34  
35      @Override
36      public boolean equals(Object o) {
37          if (this == o) return true;
38          if (!(o instanceof PreparedResponse that)) return false;
39          return Objects.equals(statusLineAndHeaders, that.statusLineAndHeaders) && Arrays.equals(body, that.body);
40      }
41  
42      @Override
43      public int hashCode() {
44          int result = Objects.hash(statusLineAndHeaders);
45          result = 31 * result + Arrays.hashCode(body);
46          return result;
47      }
48  
49      @Override
50      public String toString() {
51          return "PreparedResponse{" +
52                  "statusLineAndHeaders='" + statusLineAndHeaders + '\'' +
53                  ", body=" + Arrays.toString(body) +
54                  '}';
55      }
56  }
