87      /**
88       * This is the brains of how the server responds to web clients.  Whatever
89       * code lives here will be inserted into a slot within the server code.
90       * See {@link Server#start(ExecutorService, ThrowingConsumer)}
91       *
92       * @param handlerFinder bear with me...  This is a function, that takes a {@link RequestLine}, and
93       *                      returns a {@link Function} that handles the {@link Request} -> {@link Response}.
94       *                      Normally, you would just use {@link #makePrimaryHttpHandler()} and the default code at
95       *                      {@link #findEndpointForThisStartline(RequestLine)} would be called.  However, you can provide
96       *                      a handler here if you want to override that behavior, for example in tests when
97       *                      you want a bit more control.
98       *                      <br>
99       *                      The common case definition of this is found at {@link #findEndpointForThisStartline}
100      */
101     ThrowingConsumer<ISocketWrapper, IOException> makePrimaryHttpHandler(Function<RequestLine, Function<Request, Response>> handlerFinder) {
102 
103         // build the handler
104 
105         return sw -> {
106             try (sw) {
107 
108                 // if we recognize this client as an attacker, dump them.
109                 ITheBrig theBrig = (fs != null && fs.getTheBrig() != null) ? fs.getTheBrig() : null;
110                 if (theBrig != null) {
111                     String remoteClient = sw.getRemoteAddr();
112                     if (theBrig.isInJail(remoteClient + "_vuln_seeking")) {
113                         // if this client is a vulnerability seeker, just dump them unceremoniously
114                         logger.logDebug(() -> "closing the socket on " + remoteClient);
115                         return;
116                     }
117                 }
118 
119                 var fullStopwatch = stopWatchUtils.startTimer();
120                 final var is = sw.getInputStream();
121 
122                 /*
123                 By default, browsers expect the server to run in keep-alive mode.
124                 We'll break out later if we find that the browser doesn't do keep-alive
125                  */
126                 while(true) {
127 
128                     // first grab the start line (see https://developer.mozilla.org/en-US/docs/Web/HTTP/Messages#start_line)
129                     // e.g. GET /foo HTTP/1.1
130                     final var rawStartLine = inputStreamUtils.readLine(is);
131                      /*
132                       if the rawStartLine is null, that means the client stopped talking.
133                       See ISocketWrapper.readLine()
134                       */
135                     if (rawStartLine == null) {
136                         // no need to do any further work, just bail
137                         return;
138                     }
139 
140                     logger.logTrace(() -> sw + ": raw startline received: " + rawStartLine);
141                     var sl = RequestLine.EMPTY(context).extractStartLine(rawStartLine);
142                     logger.logTrace(() -> sw + ": StartLine received: " + sl.toString());
143                     if (sl.getRawValue().isBlank()) {
144                         /*
145                         if we get in here, it means the client sent nothing in the spot
146                         where the Start Line should have been - therefore, we're not
147                         dealing with a kosher request at all. Bail.
148                          */
149                         return;
150                     }
151 
152                     String suspiciousClues = underInvestigation.isLookingForSuspiciousPaths(sl.getPathDetails().isolatedPath());
153                     if (suspiciousClues.length() > 0 && theBrig != null) {
154                         logger.logDebug(() -> sw.getRemoteAddr() + " is looking for a vulnerability, for this: " + suspiciousClues);
155                         theBrig.sendToJail(sw.getRemoteAddr() + "_vuln_seeking", constants.VULN_SEEKING_JAIL_DURATION);
156                         return;
157                     }
158 
159                     /****************************************
160                      *   Set up some important variables
161                      ***************************************/
162                     // The request we received from the client
163                     Request clientRequest = Request.EMPTY(context);
164                     // The response we will send to the client
165                     Response resultingResponse;
166                     boolean isKeepAlive = false;
167 
168                     /*
169                        next we will read the headers (e.g. Content-Type: foo/bar) one-by-one.
170 
171                        by the way, the headers will tell us vital information about the
172                        body.  If, for example, we're getting a POST and receiving a
173                        www form url encoded, there will be a header of "content-length"
174                        that will mention how many bytes to read.  On the other hand, if
175                        we're receiving a multipart, there will be no content-length, but
176                        the content-type will include the boundary string.
177                     */
178                     var headers = Headers.make(context, inputStreamUtils);
179                     Headers hi = headers.extractHeaderInformation(sw.getInputStream());
180                     logger.logTrace(() -> "The headers are: " + hi.getHeaderStrings());
181 
182                     // determine if we are in a keep-alive connection
183                     if (sl.getVersion() == HttpVersion.ONE_DOT_ZERO) {
184                         isKeepAlive = hi.hasKeepAlive();
185                     } else if (sl.getVersion() == HttpVersion.ONE_DOT_ONE) {
186                         isKeepAlive = ! hi.hasConnectionClose();
187                     }
188                     boolean finalIsKeepAlive = isKeepAlive;
189                     logger.logTrace(() -> "Is this a keep-alive connection? " + finalIsKeepAlive);
190 
191                     Body body = Body.EMPTY;
192                     // Determine whether there is a body (a block of data) in this request
193                     if (isThereIsABody(hi)) {
194                         logger.logTrace(() -> "There is a body. Content-type is " + hi.contentType());
195                         body = bodyProcessor.extractData(sw.getInputStream(), hi);
196                         Body finalBody = body;
197                         logger.logTrace(() -> "The body is: " + finalBody.asString());
198                     }
199 
200                     /*
201                     Need to postpone processing until here, even though the request may be a 404 not found,
202                     because this way we can review the headers and other aspects of the body and maybe
203                     keep the keep-alive socket open.  For example, if the user is having an entire
204                     conversation with us and asks for a favicon and we're missing it, we shouldn't
205                     abruptly close the socket for merely that.
206 
207                     Now, on the other hand, if they are looking for vulnerabilities, it's ok to dump them.
208                      */
209                     Function<Request, Response> endpoint = handlerFinder.apply(sl);
210                     if (endpoint == null) {
211                         resultingResponse = new Response(_404_NOT_FOUND);
212                     } else {
213                         var handlerStopwatch = new StopwatchUtils().startTimer();
214                         try {
215                             clientRequest = new Request(hi, sl, body, sw.getRemoteAddr());
216                             resultingResponse = endpoint.apply(clientRequest);
217                         } catch (Exception ex) {
218                             // if an error happens while running an endpoint's code, this is the
219                             // last-chance handling of that error where we return a 500 and a
220                             // random code to the client, so a developer can find the detailed
221                             // information in the logs, which have that same value.
222                             int randomNumber = randomErrorCorrelationId.nextInt();
223                             logger.logAsyncError(() -> "error while running endpoint " + endpoint + ". Code: " + randomNumber + ". Error: " + StacktraceUtils.stackTraceToString(ex));
224                             resultingResponse = new Response(_500_INTERNAL_SERVER_ERROR, "Server error: " + randomNumber, Map.of("Content-Type", "text/plain;charset=UTF-8"));
225                         }
226                         logger.logTrace(() -> String.format("handler processing of %s %s took %d millis", sw, sl, handlerStopwatch.stopTimer()));
227                     }
228 
229                     PreparedResponse preparedResponse = finalPreparation(clientRequest, resultingResponse, isKeepAlive);
230 
231                     // Here is where the bytes actually go out on the socket
232                     String response = preparedResponse.statusLineAndHeaders() + HTTP_CRLF;
233 
234                     logger.logTrace(() -> "Sending headers back: " + response);
235                     sw.send(response);
236 
237                     if (clientRequest.requestLine().getMethod() == RequestLine.Method.HEAD) {
238                         Request finalClientRequest = clientRequest;
239                         logger.logDebug(() -> "client " + finalClientRequest.remoteRequester() +
240                                 " is requesting HEAD for "+ finalClientRequest.requestLine().getPathDetails().isolatedPath() +
241                                 ".  Excluding body from response");
242                     } else {
243                         sw.send(preparedResponse.body());
244                     }
245                     logger.logTrace(() -> String.format("full processing (including communication time) of %s %s took %d millis", sw, sl, fullStopwatch.stopTimer()));
246 
247                     if (! isKeepAlive) break;
248                 }
249             }
250         };
251     }

253     /**
254      * This handler redirects all traffic to the HTTPS endpoint.
255      * <br>
256      * It is necessary to extract the target path, but that's all
257      * the help we'll give.  We're not going to extract headers or
258      * body, we'll just read the start line and then stop reading from them.
259      */
260     ThrowingConsumer<ISocketWrapper, IOException> makeRedirectHandler() {
261         return sw -> {
262             try (sw) {
263                 try (InputStream is = sw.getInputStream()) {
264 
265                     String rawStartLine = inputStreamUtils.readLine(is);
266                  /*
267                   if the rawStartLine is null, that means the client stopped talking.
268                   See ISocketWrapper.readLine()
269                   */
270                     if (rawStartLine == null) {
271                         return;
272                     }
273 
274                     var sl = RequestLine.EMPTY(context).extractStartLine(rawStartLine);
275 
276                     // just ignore all the rest of the incoming lines.  TCP is duplex -
277                     // we'll just start talking back the moment we understand the first line.
278                     String date = ZonedDateTime.now(ZoneId.of("UTC")).format(DateTimeFormatter.RFC_1123_DATE_TIME);
279                     String hostname = constants.HOST_NAME;
280                     sw.send(
281                             "HTTP/1.1 303 SEE OTHER" + HTTP_CRLF +
282                                     "Date: " + date + HTTP_CRLF +
283                                     "Server: minum" + HTTP_CRLF +
284                                     "Location: https://" + hostname + "/" + sl.getPathDetails().isolatedPath() + HTTP_CRLF +
285                                     HTTP_CRLF
286                     );
287                 }
288             }
289         };
290     }

292     /**
293      * Determine whether the headers in this HTTP message indicate that
294      * a body is available to read
295      */
296     boolean isThereIsABody(Headers hi) {
297         // if the client sent us a content-type header at all...
298         if (!hi.contentType().isBlank()) {
299             // if the content-length is greater than 0, we've got a body
300             if (hi.contentLength() > 0) return true;
301 
302             // if the transfer-encoding header is set to chunked, we have a body
303             List<String> transferEncodingHeaders = hi.valueByKey("transfer-encoding");
304             if (transferEncodingHeaders != null && transferEncodingHeaders.stream().anyMatch(x -> x.equalsIgnoreCase("chunked"))) return true;
305         }
306         // otherwise, no body we recognize
307         return false;
308     }

310     /**
311      * This is where our strongly-typed {@link Response} gets converted
312      * to a string and sent on the socket.
313      */
314     private PreparedResponse finalPreparation(Request request, Response response, boolean isKeepAlive) {
315         String date = Objects.requireNonNullElseGet(overrideForDateTime, () -> ZonedDateTime.now(ZoneId.of("UTC"))).format(DateTimeFormatter.RFC_1123_DATE_TIME);
316 
317         StringBuilder stringBuilder = new StringBuilder();
318 
319         // add the status line
320         stringBuilder.append( "HTTP/1.1 " + response.statusCode().code + " " + response.statusCode().shortDescription + HTTP_CRLF );
321 
322         // add a date-timestamp
323         stringBuilder.append("Date: " + date + HTTP_CRLF);
324 
325         // add the server name
326         stringBuilder.append( "Server: minum" + HTTP_CRLF );
327 
328         addExtraHeaders(response, stringBuilder);
329         confirmBodyHasContentType(request, response);
330         byte[] bodyBytes = potentiallyCompress(request, response, stringBuilder);
331         applyContentLength(stringBuilder, bodyBytes);
332         addKeepAliveTimeout(isKeepAlive, stringBuilder);
333 
334         return new PreparedResponse(stringBuilder.toString(), bodyBytes);
335     }

337     /**
338      * Add extra headers specified by the business logic
339      */
340     private static void addExtraHeaders(Response response, StringBuilder stringBuilder) {
341         stringBuilder.append(
342                 response.extraHeaders().entrySet().stream()
343                 .map(x -> x.getKey() + ": " + x.getValue() + HTTP_CRLF)
344                 .collect(Collectors.joining()));
345     }

347     /**
348      * If a response body exists, it needs to have a content-type specified, or throw an exception.
349      */
350     private static void confirmBodyHasContentType(Request request, Response response) {
351         // check the correctness of the content-type header versus the data length (if any data, that is)
352         boolean hasContentType = response.extraHeaders().entrySet().stream().anyMatch(x -> x.getKey().toLowerCase(Locale.ROOT).equals("content-type"));
353 
354         // if there *is* data, we had better be returning a content type
355         if (response.body().length > 0) {
356             mustBeTrue(hasContentType, "a Content-Type header must be specified in the Response object if it returns data. Response details: " + response + " Request: " + request);
357         }
358     }

371     /**
372      * The rules regarding the content-length header are byzantine.  Even in the cases
373      * where you aren't returning anything, servers can use this header to determine when the
374      * response is finished.
375      *
376      * There are a few rules when you MUST include it, MUST NOT, blah blah blah, but I'm
377      * not following that stuff too closely because this code skips a lot of the spec.
378      * For example, we don't handle OPTIONS or return any 1xx response types.
379      *
380      * See https://www.rfc-editor.org/rfc/rfc9110.html#name-content-length
381      */
382     private static void applyContentLength(StringBuilder stringBuilder, byte[] bodyBytes) {
383         stringBuilder.append("Content-Length: " + bodyBytes.length + HTTP_CRLF );
384     }

386     /**
387      * This method will examine the request headers and response content-type, and
388      * compress the outgoing data if necessary.
389      */
390     private static byte[] potentiallyCompress(Request request, Response response, StringBuilder stringBuilder) {
391         // we may make modifications to the response body at this point, specifically
392         // we may compress the data, if the client requested it.
393         // see https://developer.mozilla.org/en-US/docs/Web/HTTP/Headers/Accept-encoding
394         List<String> acceptEncoding = request.headers().valueByKey("accept-encoding");
395 
396         // regardless of whether the client requests compression in their Accept-Encoding header,
397         // if the data we're sending back is not of an appropriate type, we won't bother
398         // compressing it.  Basically, we're going to compress plain text.
399         Map.Entry<String, String> contentTypeHeader = SearchUtils.findExactlyOne(
400                 response.extraHeaders().entrySet().stream(), x -> x.getKey().equalsIgnoreCase("content-type"));
401         byte[] bodyBytes = response.body();
402         if (contentTypeHeader != null) {
403             String contentType = contentTypeHeader.getValue().toLowerCase();
404             if (contentType.contains("text/plain") ||
405                 contentType.contains("text/css") ||
406                 contentType.contains("text/html") ||
407                 contentType.contains("application/javascript") ||
408                 contentType.contains("text/javascript")) {
409                 bodyBytes = compressBodyIfRequested(response.body(), acceptEncoding, stringBuilder, 2048);
410             }
411         }
412         return bodyBytes;
413     }

415     /**
416      * This method will examine the content-encoding headers, and if "gzip" is
417      * requested by the client, we will replace the body bytes with compressed
418      * bytes, using the GZIP compression algorithm, as long as the response body
419      * is greater than minNumberBytes bytes.
420      *
421      * @param bodyBytes      the incoming bytes, not yet compressed
422      * @param acceptEncoding headers sent by the client about what compression
423      *                       algorithms will be understood.
424      * @param stringBuilder  the string we are gradually building up to send back to
425      *                       the client for the status line and headers. We'll use it
426      *                       here if we need to append a content-encoding - that is,
427      *                       if we successfully compress data as gzip.
428      * @param minNumberBytes number of bytes must be larger than this to compress.
429      */
430     static byte[] compressBodyIfRequested(byte[] bodyBytes, List<String> acceptEncoding, StringBuilder stringBuilder, int minNumberBytes) {
431         String allContentEncodingHeaders = acceptEncoding != null ? String.join(";", acceptEncoding) : "";
432         Matcher compressionMatcher = gzipAcceptEncodingPattern.matcher(allContentEncodingHeaders);
433         if (bodyBytes.length >= minNumberBytes && acceptEncoding != null && compressionMatcher.find()) {
434             stringBuilder.append("Content-Encoding: gzip" + HTTP_CRLF);
435             return gzipCompress(bodyBytes);
436         } else {
437             return bodyBytes;
438         }
439     }

453     /**
454      * Looks through the mappings of {@link MethodPath} and path to registered endpoints
455      * or the static cache and returns the appropriate one (If we
456      * do not find anything, return null)
457      */
458     Function<Request, Response> findEndpointForThisStartline(RequestLine sl) {
459         Function<Request, Response> handler;
460         logger.logTrace(() -> "Seeking a handler for " + sl);
461 
462         // first we check if there's a simple direct match
463         String requestedPath = sl.getPathDetails().isolatedPath().toLowerCase(Locale.ROOT);
464 
465         // if the user is asking for a HEAD request, they want to run a GET command
466         // but don't want the body.  We'll simply exclude sending the body, later on, when returning the data
467         RequestLine.Method method = sl.getMethod() == RequestLine.Method.HEAD ? RequestLine.Method.GET : sl.getMethod();
468 
469         MethodPath key = new MethodPath(method, requestedPath);
470         handler = registeredDynamicPaths.get(key);
471 
472         if (handler == null) {
473             logger.logTrace(() -> "No direct handler found.  looking for a partial match for " + requestedPath);
474             handler = findHandlerByPartialMatch(sl);
475         }
476 
477         if (handler == null) {
478             logger.logTrace(() -> "No partial match found, checking files on disk for " + requestedPath );
479             handler = findHandlerByFilesOnDisk(sl);
480         }
481 
482         // we'll return this, and it could be a null.
483         return handler;
484     }
485 
486     /**
487      * last ditch effort - look on disk.  This response will either
488      * be the file to return, or null if we didn't find anything.
489      */
490     private Function<Request, Response> findHandlerByFilesOnDisk(RequestLine sl) {
491         if (sl.getMethod() != RequestLine.Method.GET) {
492             return null;
493         }
494         String requestedPath = sl.getPathDetails().isolatedPath();
495         Response response = fileUtils.readStaticFile(requestedPath);
496         if (response != null) {
497             return request -> response;
498         } else {
499             return null;
500         }
501     }

503     /**
504      * let's see if we can match the registered paths against a **portion** of the startline
505      */
506     Function<Request, Response> findHandlerByPartialMatch(RequestLine sl) {
507         String requestedPath = sl.getPathDetails().isolatedPath();
508         var methodPathFunctionEntry = registeredPartialPaths.entrySet().stream()
509                 .filter(x -> requestedPath.startsWith(x.getKey().path()) &&
510                         x.getKey().method().equals(sl.getMethod()))
511                 .findFirst().orElse(null);
512         if (methodPathFunctionEntry != null) {
513             return methodPathFunctionEntry.getValue();
514         } else {
515             return null;
516         }
517     }

581         registeredPartialPaths.put(new MethodPath(method, pathName), webHandler);

584     /**
585      * This allows users to add extra mappings

594      * </p>
